79 void Sprite::create(const std::string& texture, const int layer, const float opacity)
80 {
81 auto& atlas = core::ServiceLocator<resource::TextureAtlas>::ref();
82
83 const auto info_opt = atlas.query(texture);
84 if (info_opt.has_value())
85 {
86 const auto& info = info_opt.value().get();
87
88 m_tex_name = texture;
89 m_opacity = std::clamp(opacity, 0.0f, 1.0f);
90 m_width = static_cast<float>(info.m_region.m_width);
91 m_height = static_cast<float>(info.m_region.m_height);
92 m_layer = layer;
93 m_texture_handle = info.m_handle;
94
95 std::array<graphics::Vertex, 4> vertices;
96
97 vertices[0].m_pos = {0.0f, 0.0f};
98 vertices[0].m_texels = info.m_texel_region.ul_texels();
99
100 vertices[1].m_pos = {m_width, 0.0f};
101 vertices[1].m_texels = info.m_texel_region.ur_texels();
102
103 vertices[2].m_pos = {m_width, m_height};
104 vertices[2].m_texels = info.m_texel_region.br_texels();
105
106 vertices[3].m_pos = {0.0f, m_height};
107 vertices[3].m_texels = info.m_texel_region.bl_texels();
108
109 m_vao.create(vertices, graphics::StorageFlag::STATIC_DRAW, graphics::Vertex::get_default_indices(), graphics::StorageFlag::STATIC_DRAW);
110 }
111 else
112 {
113 GALAXY_LOG(GALAXY_ERROR, "Failed to query texture atlas for '{0}'.", texture);
