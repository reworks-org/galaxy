88 {

115 void NuklearUI::end_input() const
116 {
117 if (glfw.do_input)
118 {
119 for (auto i = 0; i < glfw.text_len; ++i)
120 {
121 nk_input_unicode(&glfw.ctx, glfw.text[i]);
122 }
123
124 double x = 0.0, y = 0.0;
125 glfwGetCursorPos(glfw.win, &x, &y);
126 nk_input_motion(&glfw.ctx, static_cast<int>(x), static_cast<int>(y));
127
128 nk_input_button(&glfw.ctx,
129 NK_BUTTON_DOUBLE,
130 static_cast<int>(glfw.double_click_pos.x),
131 static_cast<int>(glfw.double_click_pos.y),
132 glfw.is_double_click_down);
133 nk_input_scroll(&glfw.ctx, glfw.scroll);
134 nk_input_end(ctx());
135 glfw.text_len = 0;
136 glfw.scroll = nk_vec2(0, 0);
137 }
