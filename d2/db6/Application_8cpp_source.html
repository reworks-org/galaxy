89 // The expression dt/1s simply converts the double-based chrono seconds
90 // into a double so it can participate in the physics computation.
91 constexpr const auto dt = std::chrono::duration<long long, std::ratio<1, 60>> {1};
92 time::dt(dt / 1.0s);
93
94 using clock = std::chrono::steady_clock;
95 using duration = decltype(clock::duration {} + dt);
96 using time_point = std::chrono::time_point<clock, duration>;
97
98 duration accum = 0s;
99 time_point prev = clock::now();
100 time_point now = clock::now();
101
102 auto updates = 0u;
103 auto frames = 0u;
104 duration perf = 0s;
105
106 while (window.is_open())
107 {
108 now = clock::now();
109 auto elapsed = now - prev;
110
111 // 250ms is the limit put in place on the frame time to cope with the spiral of death.
112 // It doesn't have to be 250ms exactly but it should be sufficiently high enough to deal with spikes in load.
113 if (elapsed > 250ms)
114 {
115 elapsed = 250ms;
116 }
117
118 prev = now;
119 accum += elapsed;
120
121 while (accum >= dt)
122 {
123 perf += dt;
124 accum -= dt;
125
126 // nui.begin_input();
127 handle_events(window, dispatcher);
128 // nui.end_input();
129 // manager.update();
130
131 updates++;
132 }
133
134 // graphics::renderer::begin();
135 // manager.render();
136 // graphics::renderer::end();
137
138 frames++;
139
140 if (perf >= 1s)
141 {
142 window.append_title(std::format("{0} | UPS: {1}, FPS: {2}", Settings::title(), updates, frames));
143
144 frames = 0;
145 updates = 0;
146 perf = 0s;

187 void App::setup_async()
188 {
189 // Configure threadpool.
190 // Use half of available cores minus 2 for audio and main thread.
191 const auto system_cores = std::thread::hardware_concurrency();
192 if (system_cores < 4)
193 {
194 GALAXY_LOG(GALAXY_WARN, "Total cores are less than 4, this is not optimal.");
195 }
196
197 const auto cores = static_cast<int>(std::floor(system_cores / 2.0) - 2);
198 entt::locator<BS::light_thread_pool>::emplace(cores);
