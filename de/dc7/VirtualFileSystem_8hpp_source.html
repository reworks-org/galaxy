318 template<meta::vfs_read_type FileType>
319 inline FileType VirtualFileSystem::read(const std::string& file)
320 {
321 auto entry_opt = find(file);
322 if (entry_opt.has_value())
323 {
324 FileType buffer {};
325
326 auto& entry = entry_opt.value();
327
328 auto z = zip_open(entry.pack.c_str(), ZIP_DEFAULT_COMPRESSION_LEVEL, 'r');
329
330 if (zip_entry_open(z, entry.entry.c_str()) >= 0)
331 {
332 buffer.resize(zip_entry_size(z));
333
334 if (zip_entry_noallocread(z, buffer.data(), buffer.size()) < 0)
335 {
336 GALAXY_LOG(GALAXY_ERROR, "Failed to read {0} from vfs.", file);
337 }
338
339 zip_entry_close(z);
340 }
341 else
342 {
343 GALAXY_LOG(GALAXY_ERROR, "Failed to open zip entry for {0}.", file);
344 }
345
346 zip_close(z);
347
348 return buffer;
349 }
350 else
351 {
352 GALAXY_LOG(GALAXY_ERROR, "Failed to find file '{0}' in vfs.", file);
353 }
354
355 return {};
356 }

358 template<meta::vfs_write_type FileType>
359 inline bool VirtualFileSystem::write(FileType data, const std::string& file)
360 {
361 bool result = false;
362
363 if (!data.empty() && !file.empty())
364 {
365 auto entry_opt = find(file);
366 if (entry_opt.has_value())
367 {
368 // This should be copied, so its ok to reference after.
369 ArchiveEntry ae = entry_opt.value();
370 remove(file, ae);
371
372 auto z = zip_open(ae.pack.c_str(), ZIP_DEFAULT_COMPRESSION_LEVEL, 'a');
373 if (zip_entry_open(z, ae.entry.c_str()) >= 0)
374 {
375 if (zip_entry_write(z, data.data(), data.size()) < 0)
376 {
377 GALAXY_LOG(GALAXY_ERROR, "Failed to write entry to zip for file '{0}'.", file);
378 }
379 else
380 {
381 result = true;
382 }
383
384 zip_entry_close(z);
385 }
386 else
387 {
388 GALAXY_LOG(GALAXY_ERROR, "Failed to open zip entry for {0}.", file);
389 }
390 }
391 else
392 {
393 auto z = zip_open(m_datapack.c_str(), ZIP_DEFAULT_COMPRESSION_LEVEL, 'w');
394 if (zip_entry_open(z, file.c_str()) == 0)
395 {
396 if (zip_entry_write(z, data.data(), data.size()) < 0)
397 {
398 GALAXY_LOG(GALAXY_ERROR, "Failed to write entry to zip for file '{0}'.", file);
399 }
400 else
401 {
402 result = true;
403 const auto path = std::filesystem::path(zip_entry_name(z));
404 m_tree[path.filename().string()] =
405 ArchiveEntry {.entry = zip_entry_name(z), .pack = m_datapack, .type = get_asset_type_from_path(path.string())};
406 }
407
408 zip_entry_close(z);
409 }
410 else
411 {
412 GALAXY_LOG(GALAXY_ERROR, "Failed to open zip entry for {0}.", file);
413 }
414 zip_close(z);
415 }
416 }
417 else
418 {
419 GALAXY_LOG(GALAXY_ERROR, "Passed empty data or filename to vfs to write.");
420 }
421
422 return result;
423 }

425 template<meta::vfs_read_type FileType>
426 inline FileType VirtualFileSystem::read_disk(const std::string& file)
427 {
428 if (!file.empty())
429 {
430 std::filesystem::path path = file;
431 if (!path.is_absolute())
432 {
433 path = GALAXY_ROOT_DIR / path;
434 }
435
436 std::ifstream ifs;
437
438 if constexpr (std::is_same_v<FileType, meta::FSTextR>)
439 {
440 ifs.open(path.string(), std::ifstream::in);
441 }
442 else if constexpr (std::is_same_v<FileType, meta::FSBinaryR>)
443 {
444 ifs.open(path.string(), std::ifstream::in | std::ifstream::binary | std::ifstream::ate);
445 }
446
447 if (ifs.good())
448 {
449 if constexpr (std::is_same_v<FileType, meta::FSTextR>)
450 {
451 std::stringstream buffer;
452 buffer << ifs.rdbuf();
453 ifs.close();
454
455 return buffer.str();
456 }
457 else if constexpr (std::is_same_v<FileType, meta::FSBinaryR>)
458 {
459 meta::vector<char> buffer;
460
461 const auto size = ifs.tellg();
462 buffer.resize(size);
463
464 ifs.seekg(0, std::ifstream::beg);
465 ifs.read(&buffer[0], size);
466 ifs.close();
467
468 meta::FSBinaryR out;
469 reinterpret_cast<meta::vector<char>&>(out) = std::move(buffer);
470
471 return out;
472 }
473 }
474 else
475 {
476 ifs.close();
477 GALAXY_LOG(GALAXY_ERROR, "Failed to read file '{0}'.", file);
478 }
479 }
480 else
481 {
482 GALAXY_LOG(GALAXY_ERROR, "Tried to read empty file from disk.");
483 }
484
485 return {};
486 }

488 template<meta::vfs_write_type FileType>
489 inline bool VirtualFileSystem::write_disk(FileType data, const std::string& file)
490 {
491 if (!data.empty() && !file.empty())
492 {
493 std::filesystem::path path = file;
494 if (!path.is_absolute())
495 {
496 path = GALAXY_ROOT_DIR / path;
497 }
498
499 std::ofstream ofs;
500
501 if constexpr (std::is_same_v<FileType, meta::FSTextR>)
502 {
503 ofs.open(path.string(), std::ofstream::out | std::ofstream::trunc);
504 }
505 else if constexpr (std::is_same_v<FileType, meta::FSBinaryR>)
506 {
507 ofs.open(path.string(), std::ofstream::out | std::ofstream::trunc | std::ofstream::binary);
508 }
509
510 if (ofs.good())
511 {
512 if constexpr (std::is_same_v<FileType, meta::FSTextR>)
513 {
514 ofs << data;
515 }
516 else if constexpr (std::is_same_v<FileType, meta::FSBinaryR>)
517 {
518 ofs.write(data.data(), data.size());
519 }
520
521 ofs.close();
522 return true;
523 }
524 else
525 {
526 ofs.close();
527 GALAXY_LOG(GALAXY_ERROR, "Failed to save file data to disk '{0}'.", file);
528 }
529 }
530 else
531 {
532 GALAXY_LOG(GALAXY_ERROR, "Tried to write incorrectly to disk.");
533 }
534
535 return false;
