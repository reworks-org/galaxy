285 void Application::run()
286 {
287#ifdef NDEBUG
288 try
289 {
290#endif
291 GALAXY_CUR_UPS = 0;
292 GALAXY_CUR_FPS = 0;
293
294 auto& config = ServiceLocator<Config>::ref();
295 auto& window = ServiceLocator<Window>::ref();
296 auto& manager = ServiceLocator<scene::SceneManager>::ref();
297 auto& nui = ServiceLocator<ui::NuklearUI>::ref();
298 auto& renderer = ServiceLocator<graphics::Renderer>::ref();
299
300 const auto log_perf = config.get<bool>("log_performance");
301
302 unsigned int frames = 0u;
303 unsigned int updates = 0u;
304
305 constexpr const auto ups_as_nano = std::chrono::duration_cast<std::chrono::nanoseconds>(GALAXY_UPS);
306 constexpr const auto one_second = std::chrono::seconds {1};
307
308 std::chrono::nanoseconds accumulator {0};
309 std::chrono::nanoseconds perf_counter {0};
310 std::chrono::nanoseconds elapsed {0};
311
312 auto current = std::chrono::high_resolution_clock::now();
313 auto previous = current;
314
315 while (window.is_open())
316 {
317 current = std::chrono::high_resolution_clock::now();
318 elapsed = current - previous;
319 previous = current;
320 accumulator += elapsed;
321
322 auto& scene = manager.current();
323
324 perf_counter += elapsed;
325
326 while (accumulator >= GALAXY_UPS)
327 {
328 renderer.flush();
329
330 window.poll_events();
331 scene.update();
332
333 accumulator -= ups_as_nano;
334
335 updates++;
336 }
337
338 scene.render();
339 renderer.swap_buffers();
340
341 frames++;
342
343 if (perf_counter >= one_second)
344 {
345 if (log_perf)
346 {
347 GALAXY_LOG(GALAXY_INFO, "FPS: {0} | UPS: {1}.", frames, updates);
348 }
349
350 GALAXY_CUR_UPS = updates;
351 GALAXY_CUR_FPS = frames;
352
353 frames = 0;
354 updates = 0;
355 perf_counter = std::chrono::nanoseconds {0};
356 }
357 }
358#ifdef NDEBUG
359 }
360 catch (const std::exception& e)
361 {
362 GALAXY_LOG(GALAXY_ERROR, "Main loop exception: '{0}'.", e.what());
363 }
364#endif
365 }
