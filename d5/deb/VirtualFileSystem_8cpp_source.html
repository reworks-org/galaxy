87 {
88 PHYSFS_File* f = PHYSFS_openRead(file.c_str());
89 if (log::physfs_check(f))
90 {
91 const auto len = PHYSFS_fileLength(f);
92
93 if (log::physfs_check(len))
94 {
95 std::string buffer;

104 GALAXY_LOG(GALAXY_ERROR, "Failed to read '{0}'.", file);
