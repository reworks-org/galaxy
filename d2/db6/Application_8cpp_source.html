297 void App::run()
298 {
299 GALAXY_CUR_UPS = 0;
300 GALAXY_CUR_FPS = 0;
301
302 auto& window = ServiceLocator<Window>::ref();
303 auto& manager = ServiceLocator<scene::SceneManager>::ref();
304 auto& nui = ServiceLocator<ui::NuklearUI>::ref();
305
306 unsigned int frames = 0u;
307 unsigned int updates = 0u;
308
309 constexpr const auto ups_as_nano = std::chrono::duration_cast<std::chrono::nanoseconds>(GALAXY_UPS);
310 constexpr const auto one_second = std::chrono::seconds {1};
311
312 std::chrono::nanoseconds accumulator {0};
313 std::chrono::nanoseconds perf_counter {0};
314 std::chrono::nanoseconds elapsed {0};
315
316 auto current = std::chrono::high_resolution_clock::now();
317 auto previous = current;
318
319 while (window.is_open())
320 {
321 current = std::chrono::high_resolution_clock::now();
322 elapsed = current - previous;
323 previous = current;
324 accumulator += elapsed;
325
326 perf_counter += elapsed;
327
328 while (accumulator >= GALAXY_UPS)
329 {
330 nui.begin_input();
331 window.poll_events();
332 nui.end_input();
333 manager.update();
334
335 accumulator -= ups_as_nano;
336 updates++;
337 }
338
339 manager.render();
340
341 frames++;
342
343 if (perf_counter >= one_second)
344 {
345 GALAXY_CUR_UPS = updates;
346 GALAXY_CUR_FPS = frames;
347 window.append_title(std::format(" | UPS: {0}, FPS: {1}", updates, frames));
348
349 frames = 0;
350 updates = 0;
351 perf_counter = std::chrono::nanoseconds {0};
352 }
353 }
