75 {
76 auto& fs = core::ServiceLocator<fs::VirtualFileSystem>::ref();

78 auto path = std::filesystem::path(filepath);
79 auto full = GALAXY_ROOT_DIR / GALAXY_WORK_DIR / path.parent_path() / path.stem();

81 if (!std::filesystem::exists(full.parent_path()))
82 {
83 std::filesystem::create_directories(full.parent_path());
84 }
85
