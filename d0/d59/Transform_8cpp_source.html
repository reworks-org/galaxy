152 glm::mat4 Transform::get_transform()
153 {
154 auto rotation = glm::translate(identity_matrix, m_origin);
155 rotation = glm::rotate(rotation, glm::radians(m_rotation), rotation_vec);
156 rotation = glm::translate(rotation, -m_origin);
157
158 auto scale = glm::translate(identity_matrix, m_origin);
159 scale = glm::scale(scale, {m_scale, 1.0f});
160 scale = glm::translate(scale, -m_origin);
161
162 return glm::translate(identity_matrix, m_pos) * rotation * scale;
163 }
