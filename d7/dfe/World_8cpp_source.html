120 void World::update()
121 {
122 for (const auto& [rigidbody, transform] : m_bodies_to_construct)
123 {
124 b2BodyDef def {};
125 def.position.x = transform->get_pos().x / GALAXY_WORLD_TO_BOX;
126 def.position.y = transform->get_pos().y / GALAXY_WORLD_TO_BOX;
127 def.angle = glm::radians(transform->get_rotation());
128 def.type = rigidbody->m_type;
129 def.fixedRotation = rigidbody->m_fixed_rotation;
130 def.bullet = rigidbody->m_bullet;
131
132 rigidbody->m_body = m_b2world->CreateBody(&def);
133
134 b2PolygonShape shape;
135 shape.SetAsBox(rigidbody->m_shape.x, rigidbody->m_shape.y);
136
137 b2FixtureDef fixture;
138 fixture.shape = &shape;
139 fixture.density = rigidbody->m_density;
140 fixture.friction = rigidbody->m_friction;
141 fixture.restitution = rigidbody->m_restitution;
142 fixture.restitutionThreshold = rigidbody->m_restitution_threshold;
143
144 rigidbody->m_body->CreateFixture(&fixture);
145 }
146
147 m_bodies_to_construct.clear();
148 m_b2world->Step(GALAXY_DT, m_velocity_iterations, m_position_iterations);
149
150 for (auto i = 0; i < m_systems.size(); i++)
151 {
152 m_systems[i]->update(m_scene);
153 }
154 }

268 void World::construct_script(entt::registry& registry, entt::entity entity)
269 {
270 auto& script = registry.get<components::Script>(entity);
271 auto& state = core::ServiceLocator<sol::state>::ref();
272
273 auto result = state.load_file(script.file());
274 if (result.valid())
275 {
276 script.m_self = result.call();
277
278 if (script.m_self.valid())
279 {
280 script.m_update = script.m_self["update"];
281
282 if (!script.m_update.valid())
283 {
284 GALAXY_LOG(GALAXY_ERROR, "Update function not present in '{0}'.", script.file());
285 }
286
287 script.m_self["owner"] = std::ref(registry);
288 script.m_self["dispatcher"] = std::ref(m_scene->m_dispatcher);
289 script.m_self["id"] = sol::readonly_property([entity] {
290 return entity;
291 });
292
293 sol::function init = script.m_self["construct"];
294 if (init.valid())
295 {
296 init(script.m_self);
297 }
298 }
299 else
300 {
301 GALAXY_LOG(GALAXY_ERROR, "Failed to validate script '{0}'. Make sure its in the correct format.", script.file());
302 }
303 }
304 else
305 {
306 GALAXY_LOG(GALAXY_ERROR, "Failed to load script '{0}' because '{1}'.", script.file(), magic_enum::enum_name(result.status()));
307 }
