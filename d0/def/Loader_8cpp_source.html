40 void Loader::extract_assets()
41 {
42 auto& config = ServiceLocator<Config>::ref();
43 auto& nui = ServiceLocator<ui::NuklearUI>::ref();
44 auto& renderer = ServiceLocator<graphics::Renderer>::ref();
45
46 const auto asset_dir = config.get<std::string>("asset_dir");
47 const auto compressed_assets = config.get<std::string>("compressed_assets");
48
49 // Unpack assets if zip is present.
50 if (std::filesystem::exists(compressed_assets))
51 {
52 GALAXY_LOG(GALAXY_INFO, "Detected asset pack. Extracting...");
53 GALAXY_LOG(GALAXY_INFO, "Removing old assets...");
54 std::filesystem::remove_all(asset_dir);
55
56 // Get file count.
57 auto z = zip_open(compressed_assets.c_str(), 0, 'r');
58 int count = static_cast<int>(zip_entries_total(z));
59 zip_close(z);
60
61 GALAXY_LOG(GALAXY_INFO, "Beginning asset extraction...");
62
63 renderer.prepare_default();
64 renderer.clear();
65
66 auto extractor = [](const char* filename, void* arg) {
67 glfwPollEvents();
68
69 auto& nui = ServiceLocator<ui::NuklearUI>::ref();
70 auto& renderer = ServiceLocator<graphics::Renderer>::ref();
71
72 static int i = 0;
73 const int n = *static_cast<int*>(arg);
74
75 ++i;
76 nui.new_frame();
77 nui.show_loading_bar("Extracting Assets...", i, n);
78
79 renderer.clear();
80 nui.render();
81 renderer.swap_buffers();
82
83 return 0;
84 };
85
86 if (zip_extract(compressed_assets.c_str(), "./", extractor, &count) == 0)
87 {
88 std::filesystem::remove(compressed_assets);
89 GALAXY_LOG(GALAXY_INFO, "Successfully extracted assets.");
90 }
91 else
92 {
93 std::filesystem::remove_all(asset_dir);
94 GALAXY_LOG(GALAXY_FATAL, "Failed to extract assets.");
95 }
96 }
97
98 // Clear popup.
99 renderer.clear();
100 renderer.swap_buffers();
101 }

238
239 auto icon = config.get<std::string>("icon", "window");

242 window.set_icon(icon);
243 }
244
245 if (config.get<bool>("allow_native_closing", "window"))
246 {
247 window.allow_native_closing();
248 }
249 else
250 {
251 window.prevent_native_closing();
252 }
253
254 auto& cursor = window.get_input<input::Cursor>();
255 cursor.toggle(config.get<bool>("visible_cursor", "window"));

257 auto cursor_icon = config.get<std::string>("cursor_icon", "window");
258 if (!cursor_icon.empty())
259 {

262 }
