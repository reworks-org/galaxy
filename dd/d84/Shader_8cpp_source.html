138 void Shader::compile()
139 {
140 // Error reporting for OpenGL.
141 char info[1024] = {0};
142 auto success = 0;
143 auto v_id = 0u;
144 auto f_id = 0u;
145
146 // Then we need to convert the stream to a c string because OpenGL requires a refernece to a c string.
147 auto v_src = m_vertex_src.c_str();
148 auto f_src = m_fragment_src.c_str();
149
150 // Retrieve the ids from opengl when creating the shader, then compile shaders, while checking for errors.
151 v_id = glCreateShader(GL_VERTEX_SHADER);
152 glShaderSource(v_id, 1, &v_src, nullptr);
153 glCompileShader(v_id);
154
155 glGetShaderiv(v_id, GL_COMPILE_STATUS, &success);
156 if (!success)
157 {
158 glGetShaderInfoLog(v_id, 1024, nullptr, info);
159
160 GALAXY_LOG(GALAXY_ERROR, "Failed to compile vertex shader. {0}.", info);
161 }
162
163 // Now do the same for the fragment shader.
164 f_id = glCreateShader(GL_FRAGMENT_SHADER);
165 glShaderSource(f_id, 1, &f_src, nullptr);
166 glCompileShader(f_id);
167
168 glGetShaderiv(f_id, GL_COMPILE_STATUS, &success);
169 if (!success)
170 {
171 glGetShaderInfoLog(f_id, 1024, nullptr, info);
172
173 GALAXY_LOG(GALAXY_ERROR, "Failed to compile fragment shader. {0}.", info);
174 }
175
176 if (m_vertex_src.empty() || m_fragment_src.empty())
177 {
178 success = 0;
179 }
180
181 if (success)
182 {
183 // Create and link program.
184 m_id = glCreateProgram();
185 glAttachShader(m_id, v_id);
186 glAttachShader(m_id, f_id);
187 glLinkProgram(m_id);
188
189 glGetProgramiv(m_id, GL_LINK_STATUS, &success);
190 if (!success)
191 {
192 glGetProgramInfoLog(m_id, 1024, nullptr, info);
193
194 GALAXY_LOG(GALAXY_ERROR, "Failed to attach shaders: {0}.", info);
195 }
196
197 GLint uniform_count = 0;
198 glGetProgramiv(m_id, GL_ACTIVE_UNIFORMS, &uniform_count);
199
200 if (uniform_count != 0)
201 {
202 GLint max_name_len = 0;
203 glGetProgramiv(m_id, GL_ACTIVE_UNIFORM_MAX_LENGTH, &max_name_len);
204
205 GLsizei length = 0;
206 GLsizei count = 0;
207 GLenum type = GL_NONE;
208 for (GLint i = 0; i < uniform_count; ++i)
209 {
210 auto uniform_name = std::make_unique<char[]>(max_name_len);
211 glGetActiveUniform(m_id, i, max_name_len, &length, &count, &type, uniform_name.get());
212
213 // clang-format off
214 auto uniform_info = UniformInfo
215 {
216 .location = glGetUniformLocation(m_id, uniform_name.get()),
217 .count = count
218 };
219 // clang-format on
220
221 m_cache.emplace(std::string(uniform_name.get(), length), uniform_info);
222 }
223 }
224 }
225
226 // Cleanup shaders.
227 glDeleteShader(v_id);
228 glDeleteShader(f_id);
229
230 std::string().swap(m_fragment_src);
231 std::string().swap(m_vertex_src);
232 }

297 std::optional<std::pair<std::string, std::string>> Shader::preprocess(const std::string& src)
298 {
299 std::pair<std::string, std::string> source;
300
301 const auto token = "#type";
302 const auto len = std::strlen(token);
303
304 auto pos = src.find(token, 0);
305 while (pos != std::string::npos)
306 {
307 const auto eol = src.find_first_of("\r\n", pos);
308 const auto begin = pos + len + 1;
309
310 auto type = src.substr(begin, eol - begin);
311
312 if (type != "vertex" && type != "fragment")
313 {
314 GALAXY_LOG(GALAXY_ERROR, "Failed to parse shader type. Must be 'vertex' or 'fragment'.");
315 return std::nullopt;
316 }
317 else
318 {
319 const auto next_line = src.find_first_not_of("\r\n", eol);
320 pos = src.find(token, next_line);
321
322 if (type == "vertex")
323 {
324 source.first = (pos == std::string::npos) ? src.substr(next_line) : src.substr(next_line, pos - next_line);
325 }
326 else
327 {
328 source.second = (pos == std::string::npos) ? src.substr(next_line) : src.substr(next_line, pos - next_line);
329 }
330 }
331 }
332
333 if (source.first.empty() || source.second.empty())
334 {
335 GALAXY_LOG(GALAXY_ERROR, "Failed to parse a combined shader.");
336 return std::nullopt;
337 }
338 else
339 {
340 return std::make_optional(source);
341 }
