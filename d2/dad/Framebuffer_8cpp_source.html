119 void Framebuffer::add_colour_attachment(const bool high_precision)
120 {
121 if (m_used_attachments.size() < m_max_attachments)
122 {
123 auto& config = core::ServiceLocator<core::Config>::ref();
124
125 GLuint texture = 0;
126 GLenum gl_type = (high_precision == true) ? GL_FLOAT : GL_UNSIGNED_BYTE;
127
128 glBindFramebuffer(GL_FRAMEBUFFER, m_fbo);
129
130 glGenTextures(1, &texture);
131 glBindTexture(GL_TEXTURE_2D, texture);
132 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA8, m_width, m_height, 0, GL_RGBA, gl_type, nullptr);
133
134 if (config.get<int>("trilinear_filtering", "graphics"))
135 {
136 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
137 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
138 }
139 else
140 {
141 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
142 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
143 }
144
145 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
146 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
147 glTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MAX_ANISOTROPY, static_cast<float>(config.get<int>("ansiotrophic_filtering", "graphics")));
148
149 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0 + static_cast<GLenum>(m_used_attachments.size()), GL_TEXTURE_2D, texture, 0);
150 m_used_attachments.push_back(GL_COLOR_ATTACHMENT0 + static_cast<GLenum>(m_used_attachments.size()));
151 m_attachments.push_back(std::make_pair(texture, gl_type));
152
153 glBindFramebuffer(GL_FRAMEBUFFER, 0);
154 glBindTexture(GL_TEXTURE_2D, 0);
155 }
156 else
157 {
158 GALAXY_LOG(GALAXY_ERROR, "No attachments left avaliable.");
159 }

281 {
282 m_width = width;
283 m_height = height;
284
285 static GLint s_cur_fbo = 0;
286 glGetIntegerv(GL_FRAMEBUFFER_BINDING, &s_cur_fbo);
287 glBindFramebuffer(GL_FRAMEBUFFER, m_fbo);
288
289 for (const auto& pair : m_attachments)
290 {
291 glBindTexture(GL_TEXTURE_2D, pair.first);
292 if (pair.second == GL_INT)
293 {
294 glTexImage2D(GL_TEXTURE_2D, 0, GL_R32I, m_width, m_height, 0, GL_RED_INTEGER, GL_INT, nullptr);
295 }
296 else
297 {
298 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA8, m_width, m_height, 0, GL_RGBA, pair.second, nullptr);
299 }
300 }
301
302 glBindTexture(GL_TEXTURE_2D, 0);
303
304 for (const auto& renderbuffer : m_renderbuffers)
305 {
306 glBindRenderbuffer(GL_RENDERBUFFER, renderbuffer);
307
308 GLint format = 0;
309 glGetRenderbufferParameteriv(GL_RENDERBUFFER, GL_RENDERBUFFER_INTERNAL_FORMAT, &format);
310
311 glRenderbufferStorage(GL_RENDERBUFFER, format, m_width, m_height);
312 }
313
314 glBindRenderbuffer(GL_RENDERBUFFER, 0);
315
316 if (m_depth_attachment != 0)
317 {
318 glBindTexture(GL_TEXTURE_2D, m_depth_attachment);
319 glTexImage2D(GL_TEXTURE_2D, 0, GL_DEPTH_COMPONENT24, m_width, m_height, 0, GL_DEPTH_COMPONENT, GL_UNSIGNED_BYTE, nullptr);
320 glBindTexture(GL_TEXTURE_2D, 0);
321 }
322
323 if (m_depth_renderbuffer != 0)
324 {
325 glBindRenderbuffer(GL_RENDERBUFFER, m_depth_renderbuffer);
326 glRenderbufferStorage(GL_RENDERBUFFER, GL_DEPTH_COMPONENT, m_width, m_height);
327 glBindRenderbuffer(GL_RENDERBUFFER, 0);
328 }

434 void Framebuffer::add_drawbuffers()
435 {
436 glDrawBuffers(static_cast<GLsizei>(m_used_attachments.size()), m_used_attachments.data());
437
438 const auto status = glCheckFramebufferStatus(GL_FRAMEBUFFER);
439 if (status != GL_FRAMEBUFFER_COMPLETE)
440 {
441 std::string reason;
442 switch (status)
443 {
444 case GL_FRAMEBUFFER_UNDEFINED:
445 reason = "GL_FRAMEBUFFER_UNDEFINED";
446 break;
447 case GL_FRAMEBUFFER_INCOMPLETE_ATTACHMENT:
448 reason = "GL_FRAMEBUFFER_INCOMPLETE_ATTACHMENT";
449 break;
450 case GL_FRAMEBUFFER_INCOMPLETE_MISSING_ATTACHMENT:
451 reason = "GL_FRAMEBUFFER_INCOMPLETE_MISSING_ATTACHMENT";
452 break;
453 case GL_FRAMEBUFFER_INCOMPLETE_DRAW_BUFFER:
454 reason = "GL_FRAMEBUFFER_INCOMPLETE_DRAW_BUFFER";
455 break;
456 case GL_FRAMEBUFFER_INCOMPLETE_READ_BUFFER:
457 reason = "GL_FRAMEBUFFER_INCOMPLETE_READ_BUFFER";
458 break;
459 case GL_FRAMEBUFFER_UNSUPPORTED:
460 reason = "GL_FRAMEBUFFER_UNSUPPORTED";
461 break;
462 case GL_FRAMEBUFFER_INCOMPLETE_MULTISAMPLE:
463 reason = "GL_FRAMEBUFFER_INCOMPLETE_MULTISAMPLE";
464 break;
465 case GL_FRAMEBUFFER_INCOMPLETE_LAYER_TARGETS:
466 reason = "GL_FRAMEBUFFER_INCOMPLETE_LAYER_TARGETS";
467 break;
468 default:
469 reason = "UNKNOWN";
470 break;
471 }
472
473 GALAXY_LOG(GALAXY_FATAL, "Failed to complete framebuffer: {0}.", reason);
474 }
475
476 glClearColor(m_clear_colour[0], m_clear_colour[1], m_clear_colour[2], m_clear_colour[3]);
477 glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
478 }
