507 void Window::set_icon(std::string_view icon)
508 {
509 auto& fs = ServiceLocator<fs::VirtualFileSystem>::ref();
510
511 const auto info = fs.find(icon);
512 if (info.code == fs::FileCode::FOUND)
513 {
514 // Fill glfw-compatible struct.
515
516 stbi_set_flip_vertically_on_load(true);
517
518 GLFWimage img = {};
519 img.pixels = stbi_load(info.string.c_str(), &img.width, &img.height, nullptr, STBI_rgb_alpha);
520
521 if (img.pixels)
522 {
523 // Copies data so safe to destroy.
524 glfwSetWindowIcon(m_window, 1, &img);
525 }
526 else
527 {
528 GALAXY_LOG(GALAXY_ERROR, "Failed to load image '{0}' for window icon.", icon);
529 }
530
531 stbi_image_free(img.pixels);
532 }
533 else
534 {
535 GALAXY_LOG(GALAXY_ERROR, "Failed to find '{0}' to use as a window icon, because '{1}'.", icon, magic_enum::enum_name(info.code));
536 }
537 }
