145  void World::update()
146  {
147  for (const auto& [rigidbody, transform] : m_bodies_to_construct)
148  {
149  b2BodyDef def {};
150  def.position.x = transform->get_pos().x / m_pixels_per_meter;
151  def.position.y = transform->get_pos().y / m_pixels_per_meter;
152  def.angle = glm::radians(transform->get_rotation());
153  def.type = rigidbody->m_type;
154  def.fixedRotation = rigidbody->m_fixed_rotation;
155  def.bullet = rigidbody->m_bullet;
156 
157  rigidbody->m_body = m_b2world->CreateBody(&def);
158 
159  b2PolygonShape shape;
160  shape.SetAsBox(rigidbody->m_shape.x, rigidbody->m_shape.y);
161 
162  b2FixtureDef fixture;
163  fixture.shape = &shape;
164  fixture.density = rigidbody->m_density;
165  fixture.friction = rigidbody->m_friction;
166  fixture.restitution = rigidbody->m_restitution;
167  fixture.restitutionThreshold = rigidbody->m_restitution_threshold;
168 
169  rigidbody->m_body->CreateFixture(&fixture);
170  }
171 
172  m_bodies_to_construct.clear();
173  m_b2world->Step(GALAXY_DT, m_velocity_iterations, m_position_iterations);
174 
175  for (auto i = 0; i < m_systems.size(); i++)
176  {
177  m_systems[i]->update(m_scene);
178  }
