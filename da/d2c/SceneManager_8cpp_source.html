240 void SceneManager::deserialize(const nlohmann::json& json)
241 {
242 m_order.clear();
243 m_scenes.clear();
244
245 const auto& scenes = json.at("scenes");
246 const auto& order = json.at("order");
247
248 m_scenes.reserve(scenes.size());
249 for (const auto& [name, data] : scenes.items())
250 {
251 auto scene = add(name);
252 if (scene)
253 {
254 scene->deserialize(data);
255 }
256 }
257
258 m_order.reserve(order.size());
259 for (const auto& [index, name] : order.items())
260 {
261 const auto name_str = name.get<std::string>();
262 m_order.insert(m_order.begin() + std::stoi(index), m_scenes[math::fnv1a_64(name_str.c_str())]);
263 }
