205 inline void Registry::construct_rendercommand(entt::registry& registry, entt::entity entity)
206 {
207 auto& cmd = registry.emplace_or_replace<components::RenderCommand>(entity).m_command;
208
209 if constexpr ((std::is_same<GraphicsComponent, components::Circle>::value) || (std::is_same<GraphicsComponent, components::Ellipse>::value) ||
210 (std::is_same<GraphicsComponent, components::Point>::value) || (std::is_same<GraphicsComponent, components::Polygon>::value) ||
211 (std::is_same<GraphicsComponent, components::Polyline>::value))
212 {
213 graphics::Shape* shape = &registry.get<GraphicsComponent>(entity).m_shape;
214
215 cmd.count = shape->vao().count();
216 cmd.instances = shape->vao().instances();
217 cmd.mode = shape->mode();
218 cmd.offset = shape->vao().offset();
219 cmd.vao = shape->vao().id();
220 cmd.uniforms.colour = shape->m_colour.vec4();
221 cmd.uniforms.entity = static_cast<int>(entt::to_integral(entity));
222 cmd.uniforms.handle = 0;
223 cmd.uniforms.textured = false;
224 cmd.uniforms.point = shape->mode() == GL_POINTS ? true : false;
225 }
226 else if constexpr (std::is_same<GraphicsComponent, components::Sprite>::value)
227 {
228 auto& sprite = registry.get<components::Sprite>(entity);
229
230 cmd.count = sprite.m_vao.count();
231 cmd.instances = sprite.m_vao.instances();
232 cmd.mode = GL_TRIANGLES;
233 cmd.offset = sprite.m_vao.offset();
234 cmd.vao = sprite.m_vao.id();
235 cmd.uniforms.colour = sprite.m_tint.vec4();
236 cmd.uniforms.entity = static_cast<int>(entt::to_integral(entity));
237 cmd.uniforms.handle = sprite.get_texture()->handle();
238 cmd.uniforms.textured = true;
239 cmd.uniforms.point = false;
240 }
241 else if constexpr (std::is_same<GraphicsComponent, components::Text>::value)
242 {
243 auto& text = registry.get<components::Text>(entity);
244
245 cmd.count = text.m_text.vao().count();
246 cmd.instances = text.m_text.vao().instances();
