186 template<valid_component ToValidate, valid_component... Dependencies>
187 inline void EntityMeta::register_dependencies()
188 {
189 const auto hash = entt::type_id<ToValidate>().hash();
190 if (!m_validations.contains(hash))
191 {
192 m_validations[hash] = [this](const entt::entity entity, entt::registry& registry) -> bool {
193 const auto* component = registry.try_get<ToValidate>(entity);
194 if (component)
195 {
196 return registry.all_of<Dependencies...>(entity);
197 }
198 else
199 {
200 // If we dont have component needing to be validated, then entity is valid.
201 return true;
202 }
203 };
204
205 m_validations_to_run.push_back(hash);
206 }
207 else
208 {
209 GALAXY_LOG(GALAXY_WARNING, "Attempted to register duplicate validation of a component dependency.");
210 }
211 }

213 template<valid_component Component>
214 inline void EntityMeta::register_component(const std::string& name)
215 {
216 if (!m_json_factory.contains(name))
217 {
218 const auto hash = entt::type_id<Component>().hash();
219 m_id_to_name.emplace(hash, name);
220 m_name_to_id.emplace(name, hash);
221
222 if constexpr (std::constructible_from<Component, const nlohmann::json&>)
223 {
224 m_json_factory.emplace(name, [](const entt::entity entity, entt::registry& registry, const nlohmann::json& json) {
225 registry.emplace<Component>(entity, json);
226 });
227
228 m_json_any_factory.emplace(name, [](const nlohmann::json& json) -> entt::any {
229 return entt::make_any<Component>(json);
230 });
231 }
232
233 if constexpr (std::derived_from<Component, fs::Serializable>)
234 {
235 m_serialize_factory.emplace(name, [name](void* component) -> SerializationData {
236 return SerializationData {.name = name, .json = static_cast<Component*>(component)->serialize()};
237 });
238 }
239 }
240 else
241 {
242 GALAXY_LOG(GALAXY_WARNING, "Attempted to register duplicate component of type '{0}'.", name);
243 }
244 }
