165 void Scene::update()
166 {
167 for (const auto& [rigidbody, transform] : m_bodies_to_construct)
168 {
169 b2BodyDef def {};
170 def.position.x = transform->m_tf.get_pos().x / GALAXY_WORLD_TO_BOX;
171 def.position.y = transform->m_tf.get_pos().y / GALAXY_WORLD_TO_BOX;
172 def.angle = glm::radians(transform->m_tf.get_rotation());
173 def.type = rigidbody->m_type;
174 def.fixedRotation = rigidbody->m_fixed_rotation;
175 def.bullet = rigidbody->m_bullet;
176
177 rigidbody->m_body = m_b2world->CreateBody(&def);
178
179 b2PolygonShape shape;
180 shape.SetAsBox(rigidbody->m_shape.x, rigidbody->m_shape.y);
181
182 b2FixtureDef fixture;
183 fixture.shape = &shape;
184 fixture.density = rigidbody->m_density;
185 fixture.friction = rigidbody->m_friction;
186 fixture.restitution = rigidbody->m_restitution;
187 fixture.restitutionThreshold = rigidbody->m_restitution_threshold;
188
189 rigidbody->m_body->CreateFixture(&fixture);
190 }
191
192 m_bodies_to_construct.clear();
193 m_b2world->Step(GALAXY_DT, m_velocity_iterations, m_position_iterations);
194
195 for (auto&& system : m_systems)
196 {
197 system->update(this);
198 }
199 }

417 void Scene::construct_script(entt::registry& registry, entt::entity entity)
418 {
419 auto& script = registry.get<components::Script>(entity);
420 auto& state = core::ServiceLocator<sol::state>::ref();
421
422 auto result = state.load_file(script.file());
423 if (result.valid())
424 {
425 script.m_self = result.call();
426
427 if (script.m_self.valid())
428 {
429 script.m_update = script.m_self["update"];
430
431 if (!script.m_update.valid())
432 {
433 GALAXY_LOG(GALAXY_ERROR, "Update function not present in '{0}'.", script.file());
434 }
435
436 script.m_self["owner"] = std::ref(registry);
437 script.m_self["id"] = sol::readonly_property([entity] {
438 return entity;
439 });
440
441 sol::function init = script.m_self["construct"];
442 if (init.valid())
443 {
444 init(script.m_self);
445 }
446 }
447 else
448 {
449 GALAXY_LOG(GALAXY_ERROR, "Failed to validate script '{0}'. Make sure its in the correct format.", script.file());
450 }
451 }
452 else
453 {
454 GALAXY_LOG(GALAXY_ERROR, "Failed to load script '{0}' because '{1}'.", script.file(), magic_enum::enum_name(result.status()));
455 }
456 }
