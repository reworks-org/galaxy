153 nlohmann::json Primitive::serialize()
154 {
155 nlohmann::json json = "{}"_json;
156
157 json["shape"] = std::string(magic_enum::enum_name(m_shape));
158 json["layer"] = m_layer;
159 json["colour"] = nlohmann::json::object();
160 json["colour"]["r"] = m_colour.m_red;
161 json["colour"]["g"] = m_colour.m_green;
162 json["colour"]["b"] = m_colour.m_blue;
163 json["colour"]["a"] = m_colour.m_alpha;
164
165 switch (m_shape)
166 {
167 case graphics::Shape::CIRCLE:
168 json["radius"] = m_data.radius;
169 json["fragments"] = m_data.fragments;
170 break;
171
172 case graphics::Shape::ELLIPSE:
173 json["radii"] = nlohmann::json::object();
174 json["radii"]["x"] = m_data.radii.x;
175 json["radii"]["y"] = m_data.radii.y;
176 json["fragments"] = m_data.fragments;
177 break;
178
179 case graphics::Shape::LINE:
180 json["line_points"] = nlohmann::json::object();
181 json["line_points"]["x1"] = m_data.start_end.x;
182 json["line_points"]["y1"] = m_data.start_end.y;
183 json["line_points"]["x2"] = m_data.start_end.z;
184 json["line_points"]["y2"] = m_data.start_end.w;
185 break;
186
187 case graphics::Shape::POLYGON:
188 case graphics::Shape::POLYLINE:
189 json["points"] = nlohmann::json::array();
190 for (const auto& point : m_data.points)
191 {
192 auto obj = nlohmann::json::object();
193 obj["x"] = point.x;
194 obj["y"] = point.y;
195
196 json["points"].push_back(obj);
197 }
198 break;
199 }
200
201 return json;
202 }

204 void Primitive::deserialize(const nlohmann::json& json)
205 {
206 Primitive::PrimitiveData data;
207
208 if (json.count("radius") > 0)
209 {
210 data.radius = json.at("radius").get<float>();
211 }
212
213 if (json.count("radii") > 0)
214 {
215 const auto& radii = json.at("radii");
216 data.radii.x = radii.at("x");
217 data.radii.y = radii.at("y");
218 }
219
220 if (json.count("fragments") > 0)
221 {
222 data.fragments = json.at("fragments").get<float>();
223 }
224
225 if (json.count("line_points") > 0)
226 {
227 const auto& points = json.at("line_points");
228 data.start_end.x = points.at("x1");
229 data.start_end.y = points.at("y1");
230 data.start_end.z = points.at("x2");
231 data.start_end.w = points.at("y2");
232 }
233
234 if (json.count("points") > 0)
235 {
236 meta::vector<glm::vec2> points;
237
238 const auto& points_array = json.at("points");
239 for (const auto& point : points_array)
240 {
241 data.points.emplace_back(point.at("x"), point.at("y"));
242 }
243 }
244
245 const auto& col_json = json.at("colour");
246 const auto colour = graphics::Colour {col_json.at("r"), col_json.at("g"), col_json.at("b"), col_json.at("a")};
247
248 const int layer = json.at("layer");
249 const auto shape_opt = magic_enum::enum_cast<graphics::Shape>(json.at("shape").get<std::string>());
250 if (shape_opt.has_value())
251 {
252 const auto shape = shape_opt.value();
253 switch (shape)
254 {
255 case graphics::Shape::CIRCLE:
256 create<graphics::Shape::CIRCLE>(data, colour, layer);
257 break;
258
259 case graphics::Shape::ELLIPSE:
260 create<graphics::Shape::ELLIPSE>(data, colour, layer);
261 break;
262
263 case graphics::Shape::LINE:
264 create<graphics::Shape::LINE>(data, colour, layer);
265 break;
266
267 case graphics::Shape::POINT:
268 create<graphics::Shape::POINT>(data, colour, layer);
269 break;
270
271 case graphics::Shape::POLYGON:
272 create<graphics::Shape::POLYGON>(data, colour, layer);
273 break;
274
275 case graphics::Shape::POLYLINE:
276 create<graphics::Shape::POLYLINE>(data, colour, layer);
277 break;
278 }
279 }
280 else
281 {
282 GALAXY_LOG(GALAXY_ERROR, "Failed to load shape enum from json, make sure it matches enum name exactly, including case.");
283 }
