26 NuklearUI::NuklearUI()
27 : m_ctx {nullptr}
28 , m_atlas {nullptr}
29 , m_viewport_backup {0, 0, 0, 0}
30 {
31 auto& window = core::ServiceLocator<core::Window>::ref();
32 auto& config = core::ServiceLocator<core::Config>::ref();
33
34 m_ctx = nk_glfw3_init(window.handle(), NK_GLFW3_DEFAULT, NK_MAX_VERTEX_BUFFER, NK_MAX_ELEMENT_BUFFER);
35
36 nk_glfw3_font_stash_begin(&m_atlas);
37
38 auto dir = config.get<std::string>("ui_folder", "resource_folders") + "fonts/";
39 auto contents = core::ServiceLocator<fs::VirtualFileSystem>::ref().list_directory(dir);
40 for (std::filesystem::path file : contents)
41 {
42 m_fonts.emplace(file.stem().string(), nk_font_atlas_add_from_file(m_atlas, file.string().c_str(), 14, nullptr));
43 }
44
45 nk_glfw3_font_stash_end();
46 nk_style_default(ctx());
47 }
