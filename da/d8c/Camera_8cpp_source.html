112 void Camera::on_key_down(events::KeyDown& e)
113 {
114 if (!e.handled)
115 {
116 if (e.keycode == input::CameraKeys::FORWARD)
117 {
118 m_pos.x -= -glm::sin(glm::radians(m_rotation)) * m_translation_speed * GALAXY_DT;
119 m_pos.y -= glm::cos(glm::radians(m_rotation)) * m_translation_speed * GALAXY_DT;
120 }
121
122 if (e.keycode == input::CameraKeys::BACKWARD)
123 {
124 m_pos.x += -glm::sin(glm::radians(m_rotation)) * m_translation_speed * GALAXY_DT;
125 m_pos.y += glm::cos(glm::radians(m_rotation)) * m_translation_speed * GALAXY_DT;
126 }
127
128 if (e.keycode == input::CameraKeys::LEFT)
129 {
130 m_pos.x -= glm::cos(glm::radians(m_rotation)) * m_translation_speed * GALAXY_DT;
131 m_pos.y -= glm::sin(glm::radians(m_rotation)) * m_translation_speed * GALAXY_DT;
132 }
133
134 if (e.keycode == input::CameraKeys::RIGHT)
135 {
136 m_pos.x += glm::cos(glm::radians(m_rotation)) * m_translation_speed * GALAXY_DT;
137 m_pos.y += glm::sin(glm::radians(m_rotation)) * m_translation_speed * GALAXY_DT;
138 }
139
140 if (m_allow_rotate)
141 {
142 if (e.keycode == input::CameraKeys::ROTATE_LEFT)
143 {
144 m_rotation += m_rotation_speed * GALAXY_DT;
145 }
146
147 if (e.keycode == input::CameraKeys::ROTATE_RIGHT)
148 {
149 m_rotation -= m_rotation_speed * GALAXY_DT;
150 }
151
152 if (m_rotation > 180.0f)
153 {
154 m_rotation -= 360.0f;
155 }
156 else if (m_rotation <= -180.0f)
157 {
158 m_rotation += 360.0f;
159 }
160
161 set_rotation(m_rotation);
162 }
163
164 set_pos(m_pos.x, m_pos.y);
165
166 e.handled = true;
167 }

267 void Camera::recalculate()
268 {
269 auto rotation = glm::translate(identity_matrix, m_origin);
270 rotation = glm::rotate(rotation, glm::radians(m_rotation), rotation_origin);
271 rotation = glm::translate(rotation, -m_origin);
272
273 auto scale = glm::translate(identity_matrix, m_origin);
274 scale = glm::scale(scale, {m_zoom, m_zoom, 1.0f});
275 scale = glm::translate(scale, -m_origin);
276
277 const auto transform = glm::translate(identity_matrix, m_pos) * rotation * scale;
278 m_data.m_model_view = glm::inverse(transform);
279 }
