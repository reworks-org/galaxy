54 void SpriteBatch::init(const int max_quads)
55 {
56 if (max_quads <= 0)
57 {
58 GALAXY_LOG(GALAXY_FATAL, "Attempted to create spritebatch with 0 quads.");
59 }
60
61 meta::vector<unsigned int> indices;
62 indices.reserve(static_cast<std::size_t>(max_quads) * 6);
63
64 auto increment = 0;
65 for (auto counter = 0; counter < max_quads; counter++)
66 {
67 indices.push_back(0 + increment);
68 indices.push_back(1 + increment);
69 indices.push_back(3 + increment);
70 indices.push_back(1 + increment);
71 indices.push_back(2 + increment);
72 indices.push_back(3 + increment);
73
74 increment += 4;
75 }
76
77 m_max_bytes = max_quads * (sizeof(Vertex) * 4);
78 m_vao.create(max_quads * 4, indices, StorageFlag::DYNAMIC_DRAW);
79 m_vertices.reserve(static_cast<std::size_t>(max_quads) * 4);
80 }

82 unsigned int SpriteBatch::push(const meta::vector<Vertex>& vertices)
83 {
84 const auto size_bytes = vertices.size() * sizeof(Vertex);
85
86 if (!((m_bytes_used + size_bytes) > m_max_bytes))
87 {
88 m_bytes_used += static_cast<unsigned int>(size_bytes);
89
90 for (const auto& vertex : vertices)
91 {
92 m_vertices.push_back(vertex);
93 }
94
95 return static_cast<unsigned int>(m_vertices.size() - 4);
96 }
97 else
98 {
99 GALAXY_LOG(GALAXY_FATAL, "Attempted to upload too many quads to a spritebatch.");
100 return 0;
101 }

104 void SpriteBatch::build()
105 {
106 // Six indicies per quad. Number of quads is total vertexs / 4.
107 // The size here is always a multiple of 4 so no "uneven" division takes place.
108 if (m_vertices.size() != 0)
109 {
110 m_index_count = 6 * (static_cast<int>(m_vertices.size()) / 4);
111 }
112
113 flush();
114 m_vao.sub_buffer(0, m_vertices);
115 m_vertices.clear();
116 }
