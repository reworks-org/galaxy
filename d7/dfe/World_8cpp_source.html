178  void World::update()
179  {
180  for (const auto& [rigidbody, transform] : m_bodies_to_construct)
181  {
182  b2BodyDef def {};
183  def.position.x = transform->get_pos().x / physics::Constants::pixels_per_meter;
184  def.position.y = transform->get_pos().y / physics::Constants::pixels_per_meter;
185  def.angle = glm::radians(transform->get_rotation());
186  def.type = rigidbody->m_type;
187  def.fixedRotation = rigidbody->m_fixed_rotation;
188  def.bullet = rigidbody->m_bullet;
189 
190  rigidbody->m_body = m_b2world->CreateBody(&def);
191 
192  b2PolygonShape shape;
193  shape.SetAsBox(rigidbody->m_shape.x, rigidbody->m_shape.y);
194 
195  b2FixtureDef fixture;
196  fixture.shape = &shape;
197  fixture.density = rigidbody->m_density;
198  fixture.friction = rigidbody->m_friction;
199  fixture.restitution = rigidbody->m_restitution;
200  fixture.restitutionThreshold = rigidbody->m_restitution_threshold;
201 
202  rigidbody->m_body->CreateFixture(&fixture);
203  }
204 
205  m_bodies_to_construct.clear();
