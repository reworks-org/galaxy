114 bool TextureArray::build_from_tileset(const std::string& file, const int tile_size)
115 {
116 auto result = true;
117
118 auto& fs = core::ServiceLocator<fs::VirtualFileSystem>::ref();
119 auto& config = core::ServiceLocator<core::Config>::ref();
120
121 auto buffer = fs.read_binary(file);
122 if (!buffer.empty())
123 {
124 stbi_set_flip_vertically_on_load(true);
125 unsigned char* data =
126 stbi_load_from_memory(buffer.data(), static_cast<int>(buffer.size() * sizeof(std::uint8_t)), &m_width, &m_height, nullptr, STBI_rgb_alpha);
127
128 if (data)
129 {
130 const auto tiles_x = m_width / tile_size;
131 const auto tiles_y = m_height / tile_size;
132 const auto tile_count = tiles_x * tiles_y;
133
134 glTextureStorage3D(m_id, 1, GL_RGBA8, tile_size, tile_size, tile_count);
135
136 unsigned int tilesheet = 0;
137 glCreateTextures(GL_TEXTURE_2D, 1, &tilesheet);
138 glTextureStorage2D(tilesheet, 1, GL_RGBA8, m_width, m_height);
139 glTextureSubImage2D(tilesheet, 0, 0, 0, m_width, m_height, GL_RGBA, GL_UNSIGNED_BYTE, data);
140
141 for (auto i = 0; i < tile_count; ++i)
142 {
143 const auto x = (i % tiles_x) * tile_size;
144 const auto y = (i / tiles_x) * tile_size;
145
146 glCopyImageSubData(tilesheet, GL_TEXTURE_2D, 0, x, y, 0, m_id, GL_TEXTURE_2D_ARRAY, 0, 0, 0, i, tile_size, tile_size, 1);
147 }
148
149 glDeleteTextures(1, &tilesheet);
150 glGenerateTextureMipmap(m_id);
151
152 if (config.get<int>("trilinear_filtering", "graphics"))

154 filter(TextureFilter::TRILINEAR);

158 filter(TextureFilter::NEAREST);

163 }
