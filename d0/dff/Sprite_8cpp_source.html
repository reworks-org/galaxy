72 void Sprite::create(const std::string& texture, const int layer, const float opacity)
73 {
74 auto& atlas = core::ServiceLocator<resource::TextureAtlas>::ref();
75
76 const auto info_opt = atlas.query(texture);
77 if (info_opt.has_value())
78 {
79 const auto& info = info_opt.value().get();
80
81 m_tex_name = texture;
82 m_opacity = std::clamp(opacity, 0.0f, 1.0f);
83 m_width = static_cast<float>(info.m_region.width);
84 m_height = static_cast<float>(info.m_region.height);
85 m_layer = layer;
86 m_tex_id = info.m_handle;
87
88 std::array<graphics::Vertex, 4> vertices;
89
90 vertices[0].m_pos = {0.0f, 0.0f};
91 vertices[0].m_texels = info.m_texel_region.ul_texels();
92
93 vertices[1].m_pos = {m_width, 0.0f};
94 vertices[1].m_texels = info.m_texel_region.ur_texels();
95
96 vertices[2].m_pos = {m_width, m_height};
97 vertices[2].m_texels = info.m_texel_region.br_texels();
98
99 vertices[3].m_pos = {0.0f, m_height};
100 vertices[3].m_texels = info.m_texel_region.bl_texels();
101
102 m_vao.create(vertices, graphics::StorageFlag::STATIC_DRAW, graphics::Vertex::get_default_indices(), graphics::StorageFlag::STATIC_DRAW);
103 }
104 else
105 {
106 GALAXY_LOG(GALAXY_ERROR, "Failed to query texture atlas for '{0}'.", texture);
