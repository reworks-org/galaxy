91 std::string VirtualFileSystem::read(const std::string& file)
92 {
93 PHYSFS_File* f = PHYSFS_openRead(file.c_str());
94 if (error::physfs_check(f))
95 {
96 const auto len = PHYSFS_fileLength(f);
97
98 if (error::physfs_check(len))
99 {
100 std::string buffer;
101 buffer.resize(len);
102
103 if (error::physfs_check(PHYSFS_readBytes(f, buffer.data(), buffer.size())))
104 {
105 return buffer;
106 }
107 else
108 {
109 GALAXY_LOG(GALAXY_ERROR, "Failed to read '{0}'.", file);
110 }
111 }
112 else
113 {
114 GALAXY_LOG(GALAXY_ERROR, "Failed to check file length for '{0}'.", file);
115 }
116
117 error::physfs_check(PHYSFS_close(f));
118 }
119 else
120 {
121 GALAXY_LOG(GALAXY_ERROR, "Failed to read '{0}'.", file);
122 }
123
124 return {};
125 }
