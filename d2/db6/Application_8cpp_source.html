395  void Application::run()
396  {
397 #ifdef NDEBUG
398  try
399  {
400 #endif
401  GALAXY_CUR_UPS = 0;
402  GALAXY_CUR_FPS = 0;
403 
404  auto& config = ServiceLocator<Config>::ref();
405  auto& window = ServiceLocator<Window>::ref();
406  auto& manager = ServiceLocator<scene::SceneManager>::ref();
407 
408  const auto log_perf = config.get<bool>("log_performance");
409 
410  unsigned int frames = 0u;
411  unsigned int updates = 0u;
412 
413  constexpr const auto ups_as_nano = std::chrono::duration_cast<std::chrono::nanoseconds>(GALAXY_UPS);
414  constexpr const auto one_second = std::chrono::seconds {1};
415 
416  std::chrono::nanoseconds accumulator {0};
417  std::chrono::nanoseconds perf_counter {0};
418  std::chrono::nanoseconds elapsed {0};
419 
420  auto current = std::chrono::high_resolution_clock::now();
421  auto previous = current;
422 
423  while (window.is_open())
424  {
425  current = std::chrono::high_resolution_clock::now();
426  elapsed = current - previous;
427  previous = current;
428  accumulator += elapsed;
429 
430  auto& scene = manager.current();
431 
432  perf_counter += elapsed;
433 
434  while (accumulator >= GALAXY_UPS)
435  {
436  graphics::Renderer::flush();
437 
438  window.poll_events();
439  scene.update();
440 
441  accumulator -= ups_as_nano;
442 
443  updates++;
444  }
445 
446  window.begin();
447  scene.render();
448  window.end();
449 
450  frames++;
451 
452  if (perf_counter >= one_second)
453  {
454  if (log_perf)
455  {
456  GALAXY_LOG(GALAXY_INFO, "FPS: {0} | UPS: {1}.", frames, updates);
457  }
458 
459  GALAXY_CUR_UPS = updates;
460  GALAXY_CUR_FPS = frames;
461 
462  frames = 0;
463  updates = 0;
464  perf_counter = std::chrono::nanoseconds {0};
465  }
466  }
467 #ifdef NDEBUG
468  }
469  catch (const std::exception& e)

471  GALAXY_LOG(GALAXY_ERROR, "Main loop exception: '{0}'.", e.what());
472  }
