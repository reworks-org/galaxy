147  void World::update()
148  {
149  for (const auto& [rigidbody, transform] : m_bodies_to_construct)
150  {
151  b2BodyDef def {};
152  def.position.x = transform->get_pos().x / m_pixels_per_meter;
153  def.position.y = transform->get_pos().y / m_pixels_per_meter;
154  def.angle = glm::radians(transform->get_rotation());
155  def.type = rigidbody->m_type;
156  def.fixedRotation = rigidbody->m_fixed_rotation;
157  def.bullet = rigidbody->m_bullet;
158 
159  rigidbody->m_body = m_b2world->CreateBody(&def);
160 
161  b2PolygonShape shape;
162  shape.SetAsBox(rigidbody->m_shape.x, rigidbody->m_shape.y);
163 
164  b2FixtureDef fixture;
165  fixture.shape = &shape;
166  fixture.density = rigidbody->m_density;
167  fixture.friction = rigidbody->m_friction;
168  fixture.restitution = rigidbody->m_restitution;
169  fixture.restitutionThreshold = rigidbody->m_restitution_threshold;
170 
171  rigidbody->m_body->CreateFixture(&fixture);
172  }
173 
174  m_bodies_to_construct.clear();
175  m_b2world->Step(GALAXY_DT, m_velocity_iterations, m_position_iterations);
