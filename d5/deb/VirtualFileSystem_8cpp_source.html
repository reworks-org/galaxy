128 void VirtualFileSystem::import(const std::string& file)
129 {
130 auto path = std::filesystem::path(file);
131 if (!path.is_absolute())
132 {
133 path = std::filesystem::absolute(GALAXY_ROOT_DIR / path);
134 }
135
136 if (path.string().find(GALAXY_WORK_DIR) != std::string::npos)
137 {
138 auto fname = path.filename().string();
139 const auto opt = find(fname);
140 if (opt.has_value())
141 {
142 remove(fname, opt.value());
143 }
144
145 auto entry = path.string();
146 strutils::replace_all(entry, GALAXY_ROOT_DIR.string(), "");
147
148 auto z = zip_open(m_datapack.c_str(), ZIP_DEFAULT_COMPRESSION_LEVEL, 'a');
149 zip_entry_open(z, entry.c_str());
150 zip_entry_fwrite(z, path.string().c_str());
151
152 m_tree[fname] = ArchiveEntry {.entry = entry, .pack = m_datapack, .type = get_asset_type_from_path(path.string())};
153 zip_entry_close(z);
154 zip_close(z);
155 }
156 else
157 {
158 GALAXY_LOG(GALAXY_ERROR, "Attempted to import file outside of working directory.");
159 }
160 }

176 void VirtualFileSystem::remove(const std::string& fname, const ArchiveEntry& entry)
177 {
178 if (m_tree.contains(fname))
179 {
180 char* entries[] = {(char*)entry.entry.c_str()};
181
182 auto z = zip_open(entry.pack.c_str(), ZIP_DEFAULT_COMPRESSION_LEVEL, 'd');
183 if (zip_entries_delete(z, entries, 1) < 0)
184 {
185 GALAXY_LOG(GALAXY_ERROR, "Failed to remove '{0}' from vfs.", fname);
186 }
187 zip_close(z);
188
189 m_tree.erase(fname);
190 }

314
315 void VirtualFileSystem::append_archive(const std::string& path)
316 {
317 auto z = zip_open(path.c_str(), ZIP_DEFAULT_COMPRESSION_LEVEL, 'r');
318
319 for (auto i = 0; i < zip_entries_total(z); ++i)
320 {
321 if (zip_entry_openbyindex(z, i) >= 0)
322 {
323 if (!zip_entry_isdir(z))
324 {
325 const auto fp = std::filesystem::path(zip_entry_name(z));
326 m_tree[fp.filename().string()] = ArchiveEntry {.entry = zip_entry_name(z), .pack = path, .type = get_asset_type_from_path(fp.string())};
327 }
328
329 zip_entry_close(z);
330 }
331 else
332 {
333 GALAXY_LOG(GALAXY_ERROR, "Failed to extract an entry from '{0}'.", path);
334 }
335 }
336
337 zip_close(z);
338 }
