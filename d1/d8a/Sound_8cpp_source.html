37 void Sound::load(Sound::Type type, std::string_view file)
38 {
39 auto& ae = core::ServiceLocator<media::AudioEngine>::ref();
40 auto& fs = core::ServiceLocator<fs::VirtualFileSystem>::ref();
41
42 ma_engine* engine = nullptr;
43 unsigned int flags = MA_SOUND_FLAG_DECODE | MA_SOUND_FLAG_NO_SPATIALIZATION;
44
45 switch (type)
46 {
47 case Sound::Type::SFX:
48 engine = ae.get_sfx_engine();
49 break;
50
51 case Sound::Type::MUSIC:
52 engine = ae.get_music_engine();
53 flags |= MA_SOUND_FLAG_STREAM;
54 break;
55
56 case Sound::Type::DIALOGUE:
57 engine = ae.get_dialogue_engine();
58 flags |= MA_SOUND_FLAG_STREAM;
59 break;
60 }
61
62 auto info = fs.find(file);
63 if (info.code == fs::FileCode::FOUND)
64 {
65 const auto result = ma_sound_init_from_file(engine, info.string.c_str(), flags, nullptr, nullptr, &m_sound);
66 if (result != MA_SUCCESS)
67 {
68 GALAXY_LOG(GALAXY_ERROR, "Failed to load {0} from file.");
69 }
70 }
71 else
72 {
73 const auto name = magic_enum::enum_name(info.code);
74 GALAXY_LOG(GALAXY_ERROR, "Failed to find {0} because {1}.", file, name);
75 }
76 }
