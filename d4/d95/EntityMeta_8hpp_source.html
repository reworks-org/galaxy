200 template<valid_component ToValidate, valid_component... Dependencies>
201 inline void EntityMeta::register_dependencies()
202 {
203 const auto hash = entt::type_id<ToValidate>().hash();
204 if (!m_validations.contains(hash))
205 {
206 m_validations[hash] = [this](const entt::entity entity, entt::registry& registry) -> bool {
207 const auto* component = registry.try_get<ToValidate>(entity);
208 if (component)
209 {
210 return registry.all_of<Dependencies...>(entity);
211 }
212 else
213 {
214 // If we dont have component needing to be validated, then entity is valid.
215 return true;
216 }
217 };
218
219 m_validations_to_run.push_back(hash);
220 }
221 else
222 {
223 GALAXY_LOG(GALAXY_WARNING, "Attempted to register duplicate validation of a component dependency.");
224 }
225 }

227 template<valid_component Component>
228 inline void EntityMeta::register_component(const std::string& name)
229 {
230 if (!m_json_factory.contains(name) && !m_ptr_factory.contains(name) && !m_any_factory.contains(name))
231 {
232 const auto hash = entt::type_id<Component>().hash();
233 m_id_to_name.emplace(hash, name);
234 m_name_to_id.emplace(name, hash);
235
236 m_json_factory.emplace(name, [](const entt::entity entity, entt::registry& registry, const nlohmann::json& json) {
237 registry.emplace<Component>(entity, json);
238 });
239
240 m_ptr_factory.emplace(name, [](void* component) -> entt::any {
241 return entt::make_any<Component>(static_cast<Component*>(component));
242 });
243
244 m_any_factory.emplace(name, [](const entt::entity entity, entt::registry& registry, entt::any& any) {
245 registry.emplace<Component>(entity, static_cast<Component*>(any.data()));
246 });
247
248 m_json_any_factory.emplace(name, [](const nlohmann::json& json) -> entt::any {
249 return entt::make_any<Component>(json);
250 });
251
252 m_serialize_factory.emplace(name, [name](void* component) -> SerializationData {
253 if constexpr (std::derived_from<Component, fs::Serializable>)
254 {
255 return SerializationData {.name = name, .json = static_cast<Component*>(component)->serialize()};
256 }
257
258 return SerializationData {.name = ""};
259 });
260 }
261 else
262 {
263 GALAXY_LOG(GALAXY_WARNING, "Attempted to register duplicate component of type '{0}'.", name);
264 }
265 }
