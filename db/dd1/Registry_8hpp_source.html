185 inline void Registry::construct_rendercommand(entt::registry& registry, entt::entity entity)
186 {
187 auto& cmd = registry.emplace_or_replace<components::RenderCommand>(entity).m_command;
188
189 if constexpr ((std::is_same<GraphicsComponent, components::Circle>::value) || (std::is_same<GraphicsComponent, components::Ellipse>::value) ||
190 (std::is_same<GraphicsComponent, components::Point>::value) || (std::is_same<GraphicsComponent, components::Polygon>::value) ||
191 (std::is_same<GraphicsComponent, components::Polyline>::value))
192 {
193 graphics::Shape* shape = &registry.get<GraphicsComponent>(entity).m_shape;
194
195 cmd.count = shape->vao().count();
196 cmd.instances = shape->vao().instances();
197 cmd.mode = shape->mode();
198 cmd.offset = shape->vao().offset();
199 cmd.vao = shape->vao().id();
200 cmd.uniforms.colour = shape->m_colour.vec4();
201 cmd.uniforms.entity = static_cast<int>(entt::to_integral(entity));
202 cmd.uniforms.handle = 0;
203 cmd.uniforms.textured = false;
204 cmd.uniforms.point = shape->mode() == GL_POINTS ? true : false;
205 }
206 else if constexpr (std::is_same<GraphicsComponent, components::Sprite>::value)
207 {
208 auto& sprite = registry.get<components::Sprite>(entity);
209
210 cmd.count = sprite.m_vao.count();
211 cmd.instances = sprite.m_vao.instances();
212 cmd.mode = GL_TRIANGLES;
213 cmd.offset = sprite.m_vao.offset();
214 cmd.vao = sprite.m_vao.id();
215 cmd.uniforms.colour = sprite.m_tint.vec4();
216 cmd.uniforms.entity = static_cast<int>(entt::to_integral(entity));
217 cmd.uniforms.handle = sprite.get_texture()->handle();
218 cmd.uniforms.textured = true;
219 cmd.uniforms.point = false;
220 }
221 else if constexpr (std::is_same<GraphicsComponent, components::Text>::value)
222 {
223 auto& text = registry.get<components::Text>(entity);
224
225 cmd.count = text.m_text.vao().count();
226 cmd.instances = text.m_text.vao().instances();
227 cmd.mode = GL_TRIANGLES;
228 cmd.offset = text.m_text.vao().offset();
229 cmd.vao = text.m_text.vao().id();
230 cmd.uniforms.colour = text.m_text.m_colour.vec4();
231 cmd.uniforms.entity = static_cast<int>(entt::to_integral(entity));
232 cmd.uniforms.handle = text.m_text.render_texture().handle();
233 cmd.uniforms.textured = true;
234 cmd.uniforms.point = false;
