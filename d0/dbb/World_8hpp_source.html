265  inline void World::register_dependencies()
266  {
267  const auto index = std::type_index(typeid(ToValidate));
268  if (!m_validations.contains(index))
269  {
270  m_validations[index] = [this](const entt::entity entity) -> bool {
271  const ToValidate* component = m_registry.try_get<ToValidate>(entity);
