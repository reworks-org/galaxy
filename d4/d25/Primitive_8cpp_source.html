127 nlohmann::json Primitive::serialize()
128 {
129 nlohmann::json json = "{}"_json;
130
131 json["shape"] = std::string(magic_enum::enum_name(m_shape));
132 json["layer"] = m_layer;
133 json["colour"] = nlohmann::json::object();
134 json["colour"]["r"] = m_colour.m_red;
135 json["colour"]["g"] = m_colour.m_green;
136 json["colour"]["b"] = m_colour.m_blue;
137 json["colour"]["a"] = m_colour.m_alpha;
138
139 switch (m_shape)
140 {
141 case math::Shape::CIRCLE:
142 json["radius"] = m_data.radius;
143 json["fragments"] = m_data.fragments;
144 break;
145
146 case math::Shape::ELLIPSE:
147 json["radii"] = nlohmann::json::object();
148 json["radii"]["x"] = m_data.radii.x;
149 json["radii"]["y"] = m_data.radii.y;
150 json["fragments"] = m_data.fragments;
151 break;
152
153 case math::Shape::LINE:
154 json["line_points"] = nlohmann::json::object();
155 json["line_points"]["x1"] = m_data.start_end.x;
156 json["line_points"]["y1"] = m_data.start_end.y;
157 json["line_points"]["x2"] = m_data.start_end.z;
158 json["line_points"]["y2"] = m_data.start_end.w;
159 break;
160
161 case math::Shape::POLYGON:
162 case math::Shape::POLYLINE:
163 json["points"] = nlohmann::json::array();
164 for (const auto& point : m_data.points)
165 {
166 auto obj = nlohmann::json::object();
167 obj["x"] = point.x;
168 obj["y"] = point.y;
169
170 json["points"].push_back(obj);
171 }
172 break;
173 }
174
175 return json;
176 }

178 void Primitive::deserialize(const nlohmann::json& json)
179 {
180 Primitive::PrimitiveData data;
181
182 if (json.count("radius") > 0)
183 {
184 data.radius = json.at("radius").get<float>();
185 }
186
187 if (json.count("radii") > 0)
188 {
189 const auto& radii = json.at("radii");
190 data.radii.x = radii.at("x");
191 data.radii.y = radii.at("y");
192 }
193
194 if (json.count("fragments") > 0)
195 {
196 data.fragments = json.at("fragments").get<float>();
197 }
198
199 if (json.count("line_points") > 0)
200 {
201 const auto& points = json.at("line_points");
202 data.start_end.x = points.at("x1");
203 data.start_end.y = points.at("y1");
204 data.start_end.z = points.at("x2");
205 data.start_end.w = points.at("y2");
206 }
207
208 if (json.count("points") > 0)
209 {
210 meta::vector<glm::vec2> points;
211
212 const auto& points_array = json.at("points");
213 for (const auto& point : points_array)
214 {
215 data.points.emplace_back(point.at("x"), point.at("y"));
216 }
217 }
218
219 const auto& col_json = json.at("colour");
220 const auto colour = graphics::Colour {col_json.at("r"), col_json.at("g"), col_json.at("b"), col_json.at("a")};
221
222 const int layer = json.at("layer");
223 const auto shape_opt = magic_enum::enum_cast<math::Shape>(json.at("shape").get<std::string>());
224 if (shape_opt.has_value())
225 {
226 const auto shape = shape_opt.value();
227 switch (shape)
228 {
229 case math::Shape::CIRCLE:
230 create<math::Shape::CIRCLE>(data, colour, layer);
231 break;
232
233 case math::Shape::ELLIPSE:
234 create<math::Shape::ELLIPSE>(data, colour, layer);
235 break;
236
237 case math::Shape::LINE:
238 create<math::Shape::LINE>(data, colour, layer);
239 break;
240
241 case math::Shape::POINT:
242 create<math::Shape::POINT>(data, colour, layer);
243 break;
244
245 case math::Shape::POLYGON:
246 create<math::Shape::POLYGON>(data, colour, layer);
247 break;
248
249 case math::Shape::POLYLINE:
250 create<math::Shape::POLYLINE>(data, colour, layer);
251 break;
252 }
253 }
254 else
255 {
256 GALAXY_LOG(GALAXY_ERROR, "Failed to load shape enum from json, make sure it matches enum name exactly, including case.");
257 }
