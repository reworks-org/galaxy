40 void Sound::load(SoundType type, const std::string& file)
41 {
42 auto& fs = core::ServiceLocator<fs::VirtualFileSystem>::ref();
43
44 ma_engine* engine = nullptr;
45 unsigned int flags = MA_SOUND_FLAG_DECODE | MA_SOUND_FLAG_NO_SPATIALIZATION;
46
47 switch (type)
48 {
49 case SoundType::SFX:
50 engine = core::ServiceLocator<media::SoundEngine>::ref().get_engine();
51 break;
52
53 case SoundType::MUSIC:
54 engine = core::ServiceLocator<media::MusicEngine>::ref().get_engine();
55 flags |= MA_SOUND_FLAG_STREAM;
56 break;
57
58 case SoundType::VOICE:
59 engine = core::ServiceLocator<media::VoiceEngine>::ref().get_engine();
60 flags |= MA_SOUND_FLAG_STREAM;
61 break;
62 }
63
64 ma_sound_uninit(&m_sound);
65 ma_decoder_uninit(&m_decoder);
66 m_data.clear();
67
68 m_data = fs.read<meta::FSBinaryR>(file);
69 if (!m_data.empty())
70 {
71 if (ma_decoder_init_memory(m_data.data(), m_data.size(), nullptr, &m_decoder) != MA_SUCCESS)
72 {
73 GALAXY_LOG(GALAXY_ERROR, "Failed to read '{0}' into decoder.", file);
74 ma_decoder_uninit(&m_decoder);
75 }
76
77 if (ma_sound_init_from_data_source(engine, &m_decoder, flags, nullptr, &m_sound) != MA_SUCCESS)
78 {
79 GALAXY_LOG(GALAXY_ERROR, "Failed to read '{0}' from decoder.", file);
80 ma_sound_uninit(&m_sound);
81 }
82 }
83 else
84 {
85 GALAXY_LOG(GALAXY_ERROR, "Failed to read '{0}' from vfs", file);
86 }
87 }
