82 bool Texture::load(std::string_view file)
83 {
84 bool result = false;
85
86 auto& fs = core::ServiceLocator<fs::VirtualFileSystem>::ref();
87 auto& config = core::ServiceLocator<core::Config>::ref();
88
89 const auto file_info = fs.find(file);
90 if (file_info.code == fs::FileCode::FOUND)
91 {
92 if (file_info.path.extension() == ".png")
93 {
94 stbi_set_flip_vertically_on_load(true);
95 unsigned char* data = stbi_load(file_info.string.c_str(), &m_width, &m_height, nullptr, STBI_rgb_alpha);
96
97 if (data)
98 {
99 glTextureStorage2D(m_texture, 1, GL_RGBA8, m_width, m_height);
100 glTextureSubImage2D(m_texture, 0, 0, 0, m_width, m_height, GL_RGBA, GL_UNSIGNED_BYTE, data);
101 glGenerateTextureMipmap(m_texture);
102
103 if (config.get<int>("trilinear_filtering", "graphics"))
104 {
105 glTextureParameteri(m_texture, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
106 glTextureParameteri(m_texture, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
107 }
108 else
109 {
110 glTextureParameteri(m_texture, GL_TEXTURE_MIN_FILTER, GL_NEAREST_MIPMAP_NEAREST);
111 glTextureParameteri(m_texture, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
112 }
113
114 glTextureParameteri(m_texture, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
115 glTextureParameteri(m_texture, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
116 glTextureParameterf(m_texture, GL_TEXTURE_MAX_ANISOTROPY, static_cast<float>(config.get<int>("ansiotrophic_filtering", "graphics")));
117
118 result = true;
119 }
120 else
121 {
122 GALAXY_LOG(GALAXY_ERROR, "Failed to load texture file '{0}' because '{1}'.", file, stbi_failure_reason());
123 }
124
125 stbi_image_free(data);
126 }
127 else
128 {
129 GALAXY_LOG(GALAXY_ERROR, "Only PNG textures are supported, invalid texture '{0}'.", file);
130 }
131 }
132 else
133 {
134 GALAXY_LOG(GALAXY_ERROR, "Failed to find texture from file '{0}'.");
135 }
136
137 return result;
138 }

200 void Texture::save(std::string_view filepath)
201 {
202 auto& fs = core::ServiceLocator<fs::VirtualFileSystem>::ref();
203
204 auto path = std::filesystem::path(filepath);
205 auto full = (fs.root_path() / path.parent_path() / path.stem()).string();
206
207 if (!full.ends_with(".png") || !full.ends_with(".PNG"))
208 {
209 full += ".png";
210 }
211
212 const auto ui = static_cast<unsigned int>(m_width) * static_cast<unsigned int>(m_height) * 4u;
213 meta::vector<unsigned int> pixels(ui, 0);
214
215 glGetTextureImage(m_texture, 0, GL_RGBA, GL_UNSIGNED_BYTE, static_cast<GLsizei>(pixels.size()), pixels.data());
216
217 stbi_flip_vertically_on_write(true);
218 stbi_write_png(full.c_str(), m_width, m_height, 4, pixels.data(), m_width * 4);
219 }
