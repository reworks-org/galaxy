25 NuklearUI::NuklearUI()
26 : m_ctx {nullptr}
27 , m_atlas {nullptr}
28 , m_viewport_backup {0, 0, 0, 0}
29 {
30 auto& window = core::ServiceLocator<core::Window>::ref();
31 auto& config = core::ServiceLocator<core::Config>::ref();
32
33 m_ctx = nk_glfw3_init(window.handle(), NK_GLFW3_DEFAULT, NK_MAX_VERTEX_BUFFER, NK_MAX_ELEMENT_BUFFER);
34
35 nk_glfw3_font_stash_begin(&m_atlas);
36
37 auto& fs = core::ServiceLocator<fs::VirtualFileSystem>::ref();
38 for (const auto& file : fs.list(GALAXY_UI_FONTS_DIR))
39 {
40 auto data = fs.read_binary(file);
41 if (!data.empty())
42 {
43 auto font = nk_font_atlas_add_from_memory(m_atlas, data.data(), data.size(), config.get<float>("ui_font_size"), nullptr);
44 m_fonts.emplace(file, font);
45 }
46 }
47
48 nk_glfw3_font_stash_end();
49 nk_style_default(ctx());
50 }
