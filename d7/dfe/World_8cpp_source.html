137 void World::update_rendersystem()
138 {
139 // Possible to do this in parallel but overhead would cost more than perf saved.
140 auto& renderer = ServiceLocator<graphics::Renderer>::ref();
141
142 const auto spr_group = m_registry.group<components::Sprite>(entt::get<components::Transform, components::Flag>);
143 const auto prim_group = m_registry.group<components::Primitive>(entt::get<components::Transform, components::Flag>);
144 const auto text_group = m_registry.group<components::Text>(entt::get<components::Transform, components::Flag>);
145
146 graphics::RenderCommand cmd;
147
148 for (auto&& [entity, sprite, transform, flag] : spr_group.each())
149 {
150 if (flag.is_flag_set<flags::Enabled>())
151 {
152 transform.set_origin(sprite.get_width() * 0.5f, sprite.get_height() * 0.5f);
153
154 cmd.instances = 1;
155 cmd.mode = graphics::primitive_to_gl(graphics::Primitives::TRIANGLE);
156 cmd.uniform_data.entity = static_cast<int>(entt::to_integral(entity));
157 cmd.uniform_data.colour = {1.0f, 1.0f, 1.0f, sprite.get_opacity()};
158 cmd.uniform_data.transform = transform.get_transform();
159 cmd.uniform_data.point = false;
160 cmd.uniform_data.textured = true;
161 cmd.renderable = &sprite;
162
163 renderer.submit(cmd);
164 }
165 }
166
167 for (auto&& [entity, primitive, transform, flag] : prim_group.each())
168 {
169 if (flag.is_flag_set<flags::Enabled>())
170 {
171 transform.set_origin(primitive.get_width() * 0.5f, primitive.get_height() * 0.5f);
172
173 cmd.instances = 1;
174 cmd.mode = primitive.get_mode();
175 cmd.uniform_data.entity = static_cast<int>(entt::to_integral(entity));
176 cmd.uniform_data.colour = primitive.m_colour.to_vec4();
177 cmd.uniform_data.transform = transform.get_transform();
178 cmd.uniform_data.point = primitive.get_shape() == graphics::Shape::POINT;
179 cmd.uniform_data.textured = false;
180 cmd.renderable = &primitive;
181
182 renderer.submit(cmd);
183 }
184 }
185
186 for (auto&& [entity, text, transform, flag] : text_group.each())
187 {
188 if (flag.is_flag_set<flags::Enabled>())
189 {
190 transform.set_origin(text.get_width() * 0.5f, text.get_height() * 0.5f);
191
192 cmd.instances = 1;
193 cmd.mode = graphics::primitive_to_gl(graphics::Primitives::TRIANGLE);
194 cmd.uniform_data.entity = static_cast<int>(entt::to_integral(entity));
195 cmd.uniform_data.colour = text.m_colour.to_vec4();
196 cmd.uniform_data.transform = transform.get_transform();
197 cmd.uniform_data.point = false;
198 cmd.uniform_data.textured = true;
199 cmd.renderable = &text;
200
201 renderer.submit(cmd);
202 }
203 }

261 void World::construct_script(entt::registry& registry, entt::entity entity)
262 {
263 auto& script = registry.get<components::Script>(entity);
264 auto& state = core::ServiceLocator<sol::state>::ref();
265
266 auto result = state.load_file(script.file());
267 if (result.valid())
268 {
269 script.m_self = result.call();
270
271 if (script.m_self.valid())
272 {
273 script.m_update = script.m_self["update"];
274
275 if (!script.m_update.valid())
276 {
277 GALAXY_LOG(GALAXY_ERROR, "Update function not present in '{0}'.", script.file());
278 }
279
280 script.m_self["owner"] = std::ref(registry);
281 script.m_self["dispatcher"] = std::ref(m_scene->m_dispatcher);
282 script.m_self["id"] = sol::readonly_property([entity] {
283 return entity;
284 });
285
286 sol::function init = script.m_self["construct"];
287 if (init.valid())
288 {
289 init(script.m_self);
290 }
291 }
292 else
293 {
294 GALAXY_LOG(GALAXY_ERROR, "Failed to validate script '{0}'. Make sure its in the correct format.", script.file());
295 }
296 }
297 else
298 {
299 GALAXY_LOG(GALAXY_ERROR, "Failed to load script '{0}' because '{1}'.", script.file(), magic_enum::enum_name(result.status()));
300 }
301 }
