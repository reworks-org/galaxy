89 void Text::update()
90 {
91 if (!m_font_name.empty())
92 {
93 auto& fonts = core::ServiceLocator<resource::Fonts>::ref();
94 m_font = fonts.get(m_font_name);
95
96 const auto vec = m_font->get_text_size(m_text, m_size);
97 const auto width = vec.x;
98 const auto height = vec.y;
99
100 if (width > m_rt.width() || height > m_rt.height())
101 {
102 m_rt.recreate(static_cast<int>(width), static_cast<int>(height));
103 }
104
105 m_rt.bind();
106
107 std::size_t start = 0;
108 std::size_t end = m_text.find('\n');
109
110 auto proj = glm::ortho(0.0f, width, height, 0.0f, -1.0f, 1.0f);
111 auto y_off = m_font->vertical_advance(m_size);
112
113 while (end != std::string::npos)
114 {
115 const auto block = m_text.substr(start, end);
116 msdfgl_printf(0,
117 y_off,
118 static_cast<int>(m_alignment),
119 m_font->handle(),
120 m_size,
121 0xffffffff,
122 glm::value_ptr(proj),
123 static_cast<msdfgl_printf_flags>(MSDFGL_UTF8 | MSDFGL_KERNING),
124 block.c_str());
125
126 y_off += m_font->vertical_advance(m_size);
127 start = end + 1;
128 end = m_text.find('\n', start);
129 }
130
131 const auto last_block = m_text.substr(start, end);
132
133 msdfgl_printf(0,
134 y_off,
135 static_cast<int>(m_alignment),
136 m_font->handle(),
137 m_size,
138 0xffffffff,
139 glm::value_ptr(proj),
140 static_cast<msdfgl_printf_flags>(MSDFGL_UTF8 | MSDFGL_KERNING),
141 last_block.c_str());
142
143 m_rt.unbind();
144
145 auto vertices = graphics::gen_quad_vertices(width, height);
146 m_vao.sub_buffer(0, vertices);
147
148 m_rt.make_bindless();
149 }
150 else
151 {
152 GALAXY_LOG(GALAXY_ERROR, "Attempted to create text without a font.");
153 }
