112 {
113 // https://stackoverflow.com/a/59446610
114
115 auto& window = entt::locator<Window>::value();
116
117 using clock = std::chrono::high_resolution_clock;
118 using duration = std::chrono::duration<double>;
119 using time_point = std::chrono::time_point<clock, duration>;
120
121 constexpr const auto one_second = std::chrono::seconds {1};
122
123 // Number of updates per second.
124 // Same as std::ratio<1, ups>, using 1000 with std::milli, etc.
125 const auto dt = one_second / settings::ups();
126
127 duration accum {0};
128 duration elapsed {0};
129 double alpha {0.0};
130
131 time_point now = clock::now();
132 time_point previous = now;
133
134 unsigned int frames = 0u;
135 unsigned int updates = 0u;
136 duration perf_counter {0};
137
138 while (!window.should_close())
139 {
140 now = clock::now();
141 elapsed = now - previous;
142 previous = now;
143
144 accum += elapsed;
145 perf_counter += elapsed;
146
147 while (accum >= dt)
148 {
149 accum -= dt;
150 alpha = dt / one_second;
151 settings::set_delta_time(alpha);
152
153 window.poll_events();
154 // manager.update();
155
156 updates++;
157 }
158
159 // BeginDrawing();
160 // manager.render();
161 // EndDrawing();
162 window.swap_buffers();
163
164 frames++;
165
166 if (perf_counter >= one_second)
167 {
168 window.append_title(std::format(" | UPS: {0}, FPS: {1}", updates, frames));
169
170 frames = 0;
171 updates = 0;
172 perf_counter = std::chrono::nanoseconds {0};
173 }
174 }

187 {
188 // Configure logging.
189 const auto now = std::chrono::zoned_time {std::chrono::current_zone(), std::chrono::system_clock::now()}.get_local_time();
190 const std::string log_path = std::format("{0}{1}{2}", log_dir, std::format("{0:%d-%m-%Y-[%H-%M]}", now), ".log");
191 if (!std::filesystem::exists(log_dir))

193 std::filesystem::create_directory(log_dir);
194 }
195
196 GALAXY_ADD_SINK(logging::FileSink, log_path);
197 GALAXY_ADD_SINK(logging::ConsoleSink);
198 GALAXY_LOG(GALAXY_INFO, "App started.");

201 void App::setup_async()
202 {
203 // Configure threadpool.

205 auto cores = (std::thread::hardware_concurrency() / 2) - 2;
