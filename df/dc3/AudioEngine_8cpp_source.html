28 , m_callback {}
29 {
30 ma_result result = ma_log_init(nullptr, &m_log);
31 if (result != MA_SUCCESS)
32 {
33 GALAXY_LOG(GALAXY_FATAL, "Failed to initialize miniaudio logging.");
34 }
35 else
36 {
37 m_callback.onLog = [](void* pUserData, ma_uint32 level, const char* pMessage) -> void {
38 GALAXY_UNUSED(pUserData);
39
40 auto msg = std::string(pMessage);
41 strutils::replace_all(msg, "\t", " ");
42 strutils::replace_all(msg, "\n", " ");
43 strutils::trim(msg);
44 strutils::make_single_spaced(msg);
45
46 switch (level)
47 {
48 case MA_LOG_LEVEL_INFO:
49 GALAXY_LOG(GALAXY_INFO, "{0}", msg);
50 break;
51
52 case MA_LOG_LEVEL_WARNING:
53 GALAXY_LOG(GALAXY_WARNING, "{0}", msg);
54 break;
55
56 case MA_LOG_LEVEL_ERROR:
57 GALAXY_LOG(GALAXY_ERROR, "{0}", msg);
58 break;
59
60 default:
61 GALAXY_LOG(GALAXY_DEBUG, "{0}", msg);
62 break;
63 }
64 };
65
66 m_callback.pUserData = nullptr;
67
68 result = ma_log_register_callback(&m_log, m_callback);
69 if (result != MA_SUCCESS)
70 {
71 GALAXY_LOG(GALAXY_FATAL, "Failed to set miniaudio logging callback.");
72 }
73 else
74 {
75 ma_engine_config config = ma_engine_config_init();
76 config.pLog = &m_log;
