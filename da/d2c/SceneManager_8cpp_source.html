296 void SceneManager::clear(bool clear_current)
297 {
298 meta::vector<std::string> remove;
299 remove.reserve(m_scenes.size());
300
301 for (auto& [name, scene] : m_scenes)
302 {
303 if (m_current)
304 {
305 if (name != m_current->m_name || clear_current)
306 {
307 scene.reset();
308 remove.push_back(name);
309 }
310 }
311 else
312 {
313 scene.reset();
314 remove.push_back(name);
315 }
316 }
317
318 for (const auto& name : remove)
319 {
320 m_scenes.erase(name);
321 }

355 void SceneManager::deserialize(const nlohmann::json& json)
356 {
357 clear(true);
358
359 const auto& scenes = json.at("scenes");
360
361 for (const auto& [key, value] : scenes.items())
362 {
363 auto scene = make_scene(key);
364 if (auto ptr = scene.lock())
365 {
366 ptr->deserialize(value);
367 }
368 }
369
370 set_scene(json.at("current_scene"));
371 }
