140  void Shader::compile()
141  {
142  // Error reporting for OpenGL.
143  char info[1024] = {0};
144  auto success = 0;
145  auto v_id = 0u;
146  auto f_id = 0u;
147 
148  // Then we need to convert the stream to a c string because OpenGL requires a refernece to a c string.
149  auto v_src = m_vertex_src.c_str();
150  auto f_src = m_fragment_src.c_str();
151 
152  // Retrieve the ids from opengl when creating the shader, then compile shaders, while checking for errors.
153  v_id = glCreateShader(GL_VERTEX_SHADER);
154  glShaderSource(v_id, 1, &v_src, nullptr);
155  glCompileShader(v_id);
156 
157  glGetShaderiv(v_id, GL_COMPILE_STATUS, &success);
158  if (!success)
159  {
160  glGetShaderInfoLog(v_id, 1024, nullptr, info);
161 
162  GALAXY_LOG(GALAXY_ERROR, "Failed to compile vertex shader. {0}.", info);
163  }
164 
165  // Now do the same for the fragment shader.
166  f_id = glCreateShader(GL_FRAGMENT_SHADER);
167  glShaderSource(f_id, 1, &f_src, nullptr);
168  glCompileShader(f_id);
169 
170  glGetShaderiv(f_id, GL_COMPILE_STATUS, &success);
171  if (!success)
172  {
173  glGetShaderInfoLog(f_id, 1024, nullptr, info);
174 
175  GALAXY_LOG(GALAXY_ERROR, "Failed to compile fragment shader. {0}.", info);
176  }
177 
178  if (m_vertex_src.empty() || m_fragment_src.empty())
179  {
180  success = 0;
181  }
182 
183  if (success)
184  {
185  // Create and link program.
186  m_id = glCreateProgram();
187  glAttachShader(m_id, v_id);
188  glAttachShader(m_id, f_id);
189  glLinkProgram(m_id);
190 
191  glGetProgramiv(m_id, GL_LINK_STATUS, &success);
192  if (!success)
193  {
194  glGetProgramInfoLog(m_id, 1024, nullptr, info);
195 
196  GALAXY_LOG(GALAXY_ERROR, "Failed to attach shaders: {0}.", info);
197  }
198 
199  GLint uniform_count = 0;
200  glGetProgramiv(m_id, GL_ACTIVE_UNIFORMS, &uniform_count);
201 
202  if (uniform_count != 0)
203  {
204  GLint max_name_len = 0;
205  glGetProgramiv(m_id, GL_ACTIVE_UNIFORM_MAX_LENGTH, &max_name_len);
206 
207  GLsizei length = 0;
208  GLsizei count = 0;
209  GLenum type = GL_NONE;
210  for (GLint i = 0; i < uniform_count; ++i)
211  {
212  auto uniform_name = std::make_unique<char[]>(max_name_len);
213  glGetActiveUniform(m_id, i, max_name_len, &length, &count, &type, uniform_name.get());
214 
215  // clang-format off
216  auto uniform_info = UniformInfo
217  {
218  .location = glGetUniformLocation(m_id, uniform_name.get()),
219  .count = count
220  };
221  // clang-format on
222 
223  m_cache.emplace(std::string(uniform_name.get(), length), uniform_info);
224  }
225  }
226  }
227 
228  // Cleanup shaders.
229  glDeleteShader(v_id);
230  glDeleteShader(f_id);
231

233  std::string().swap(m_vertex_src);
234  }
