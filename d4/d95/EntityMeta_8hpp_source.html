184  template<valid_component ToValidate, valid_component... Dependencies>
185  inline void EntityMeta::register_dependencies()
186  {
187  const auto hash = entt::type_id<ToValidate>().hash();
188  if (!m_validations.contains(hash))
189  {
190  m_validations[hash] = [this](const entt::entity entity, entt::registry& registry) -> bool {
191  const auto* component = registry.try_get<ToValidate>(entity);
192  if (component)
193  {
194  return registry.all_of<Dependencies...>(entity);
195  }
196  else
197  {
198  // If we dont have component needing to be validated, then entity is valid.
199  return true;
200  }
201  };
202 
203  m_validations_to_run.push_back(hash);
204  }
205  else
206  {
207  GALAXY_LOG(GALAXY_WARNING, "Attempted to register duplicate validation of a component dependency.");
208  }
209  }
