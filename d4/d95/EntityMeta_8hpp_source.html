180 template<valid_component ToValidate, valid_component... Dependencies>
181 inline void EntityMeta::register_dependencies()
182 {
183 const auto hash = entt::type_id<ToValidate>().hash();
184 if (!m_validations.contains(hash))
185 {
186 m_validations[hash] = [this](const entt::entity entity, entt::registry& registry) -> bool {
187 const auto* component = registry.try_get<ToValidate>(entity);
188 if (component)
189 {
190 return registry.all_of<Dependencies...>(entity);
191 }
192 else
193 {
194 // If we dont have component needing to be validated, then entity is valid.
195 return true;
196 }
197 };
198
199 m_validations_to_run.push_back(hash);
200 }
201 else
202 {
203 GALAXY_LOG(GALAXY_WARNING, "Attempted to register duplicate validation of a component dependency.");
204 }
205 }

207 template<valid_component Component>
208 inline void EntityMeta::register_component(const std::string& name)
209 {
210 if (!m_json_factory.contains(name))
211 {
212 const auto hash = entt::type_id<Component>().hash();
213 m_id_to_name.emplace(hash, name);
214 m_name_to_id.emplace(name, hash);
215
216 m_json_factory.emplace(name, [](const entt::entity entity, entt::registry& registry, const nlohmann::json& json) {
217 registry.emplace<Component>(entity, json);
218 });
219
220 m_json_any_factory.emplace(name, [](const nlohmann::json& json) -> entt::any {
221 return entt::make_any<Component>(json);
222 });
223
224 m_serialize_factory.emplace(name, [name](void* component) -> SerializationData {
225 if constexpr (std::derived_from<Component, fs::Serializable>)
226 {
227 return SerializationData {.name = name, .json = static_cast<Component*>(component)->serialize()};
228 }
229
230 return SerializationData {};
231 });
232 }
233 else
234 {
235 GALAXY_LOG(GALAXY_WARNING, "Attempted to register duplicate component of type '{0}'.", name);
236 }
237 }
