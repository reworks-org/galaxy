354 auto& cmd = registry.emplace_or_replace<components::RenderCommand>(entity).m_command;
355
356 if constexpr ((std::is_same<GraphicsComponent, components::Circle>::value) || (std::is_same<GraphicsComponent, components::Ellipse>::value) ||
357 (std::is_same<GraphicsComponent, components::Point>::value) || (std::is_same<GraphicsComponent, components::Polygon>::value) ||
358 (std::is_same<GraphicsComponent, components::Polyline>::value))
359 {
360 graphics::Shape* shape = &registry.get<GraphicsComponent>(entity).m_shape;
361
362 cmd.count = shape->vao().count();
363 cmd.instances = shape->vao().instances();
364 cmd.mode = shape->mode();
365 cmd.offset = shape->vao().offset();
366 cmd.vao = shape->vao().id();
367 cmd.uniforms.colour = shape->m_colour.vec4();
368 cmd.uniforms.entity = static_cast<int>(entt::to_integral(entity));
369 cmd.uniforms.handle = 0;
370 cmd.uniforms.textured = false;
371 cmd.uniforms.point = shape->mode() == GL_POINTS ? true : false;
372 }
373 else if constexpr (std::is_same<GraphicsComponent, components::Sprite>::value)

375 auto& sprite = registry.get<components::Sprite>(entity);
376

378 cmd.instances = sprite.m_vao.instances();
379 cmd.mode = GL_TRIANGLES;
380 cmd.offset = sprite.m_vao.offset();
381 cmd.vao = sprite.m_vao.id();
382 cmd.uniforms.colour = sprite.m_tint.vec4();
383 cmd.uniforms.entity = static_cast<int>(entt::to_integral(entity));
384 cmd.uniforms.handle = sprite.get_texture()->handle();
385 cmd.uniforms.textured = true;
