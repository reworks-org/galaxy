271  inline void World::register_dependencies() noexcept
272  {
273  const auto index = std::type_index(typeid(ToValidate));
274  if (!m_validations.contains(index))
275  {
276  m_validations[index] = [this](const entt::entity entity) noexcept -> bool {
277  const auto component = m_registry.try_get<ToValidate>(entity);
278  if (component)
279  {
