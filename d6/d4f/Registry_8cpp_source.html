161 void Registry::update(b2World& b2World)
162 {
163 for (const auto& [rigidbody, transform] : m_bodies_to_construct)
164 {
165 b2BodyDef def {};
166 def.position.x = transform->m_tf.get_pos().x / GALAXY_WORLD_TO_BOX;
167 def.position.y = transform->m_tf.get_pos().y / GALAXY_WORLD_TO_BOX;
168 def.angle = glm::radians(transform->m_tf.get_rotation());
169 def.type = rigidbody->m_type;
170 def.fixedRotation = rigidbody->m_fixed_rotation;
171 def.bullet = rigidbody->m_bullet;
172
173 rigidbody->m_body = b2World.CreateBody(&def);
174
175 b2PolygonShape shape;
176 shape.SetAsBox(rigidbody->m_shape.x, rigidbody->m_shape.y);
177
178 b2FixtureDef fixture;
179 fixture.shape = &shape;
180 fixture.density = rigidbody->m_density;
181 fixture.friction = rigidbody->m_friction;
182 fixture.restitution = rigidbody->m_restitution;
183 fixture.restitutionThreshold = rigidbody->m_restitution_threshold;
184
185 rigidbody->m_body->CreateFixture(&fixture);
186 rigidbody->m_world = &b2World;
187 }
188
189 m_bodies_to_construct.clear();
190 }

222 void Registry::construct_script(entt::registry& registry, entt::entity entity)
223 {
224 auto& script = registry.get<components::Script>(entity);
225 auto& state = core::ServiceLocator<sol::state>::ref();
226
227 auto result = state.load_file(script.file());
228 if (result.valid())
229 {
230 script.m_self = result.call();
231
232 if (script.m_self.valid())
233 {
234 script.m_update = script.m_self["update"];
235
236 if (!script.m_update.valid())
237 {
238 GALAXY_LOG(GALAXY_ERROR, "Update function not present in '{0}'.", script.file());
239 }
240
241 script.m_self["owner"] = std::ref(registry);
242 script.m_self["id"] = sol::readonly_property([entity] {
243 return entity;
244 });
245
246 sol::function init = script.m_self["construct"];
247 if (init.valid())
248 {
249 init(script.m_self);
250 }
251 }
252 else
253 {
254 GALAXY_LOG(GALAXY_ERROR, "Failed to validate script '{0}'. Make sure its in the correct format.", script.file());
255 }
256 }
257 else
258 {
259 GALAXY_LOG(GALAXY_ERROR, "Failed to load script '{0}' because '{1}'.", script.file(), magic_enum::enum_name(result.status()));
260 }
261 }
