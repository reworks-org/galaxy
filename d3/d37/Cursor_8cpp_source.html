66 void Cursor::load_custom(const std::string& file)
67 {
68 auto& fs = core::ServiceLocator<fs::VirtualFileSystem>::ref();
69
70 auto data = fs.read_binary(file);
71 if (!data.empty())
72 {
73 stbi_set_flip_vertically_on_load(true);
74
75 // Fill glfw-compatible struct.
76 GLFWimage img = {};
77 img.pixels = stbi_load_from_memory(data.data(), static_cast<int>(data.size()), &img.width, &img.height, nullptr, STBI_rgb_alpha);
78
79 if (img.pixels)
80 {
81 // Copies data so safe to destroy.
82 m_custom = glfwCreateCursor(&img, 0, 0);
83 }
84 else
85 {
86 GALAXY_LOG(GALAXY_ERROR, "Failed to load image '{0}' with stb returning '{1}'.", file, stbi_failure_reason());
87 }
88
89 stbi_image_free(img.pixels);
90 }
91 else
92 {
93 GALAXY_LOG(GALAXY_ERROR, "Failed to read '{0}'.", file);
94 }
95 }
