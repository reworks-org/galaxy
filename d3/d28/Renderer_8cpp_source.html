64 void Renderer::draw()
65 {
66 m_prev_texture = -1;
67 m_prev_nm = -1;
68
69 // I don't think we would ever sort the amount of renderable data required to do this in parallel faster.
70 // std::execution::par maybe used when > 1000? But 1000 draw calls is too slow anyway.
71
72 std::sort(m_data.begin(), m_data.end(), [](const RenderCommand& left, const RenderCommand& right) -> bool {
73 if (left.renderable->get_layer() == right.renderable->get_layer())
74 {
75 return left.renderable->get_texture_handle() < right.renderable->get_texture_handle();
76 }
77 else
78 {
79 return left.renderable->get_layer() < right.renderable->get_layer();
80 }
81 });
82
83 m_r2d_shader.bind();
84
85 for (auto& cmd : m_data)
86 {
87 m_r2d_ubo->buffer<Render2DUniform>(0, 1, &cmd.uniform_data);
88
89 glBindVertexArray(cmd.renderable->get_vao().id());
90
91 const auto tex = cmd.renderable->get_texture_handle();
92 if (m_prev_texture != tex)
93 {
94 glBindTextureUnit(0, tex);
95 m_prev_texture = tex;
96 }
97
98 // Instances = 1 is the same as glDrawElements.
99 glDrawElementsInstanced(cmd.mode, cmd.renderable->get_vao().index_count(), GL_UNSIGNED_INT, nullptr, cmd.instances);
100 }

178 void Renderer::draw_texture(const unsigned int texture, const int width, const int height)
179 {
180 auto rtt = core::ServiceLocator<resource::Shaders>::ref().get("RenderToTexture");
181 if (rtt != nullptr)
182 {
183 VertexArray va;
184 components::Transform tf;
185
186 const auto proj = glm::ortho(0.0f, static_cast<float>(width), static_cast<float>(height), 0.0f, -1.0f, 1.0f);
187
188 auto v = graphics::Vertex::gen_quad_vertices(width, height);
189 va.create(v, graphics::StorageFlag::STATIC_DRAW, graphics::Vertex::get_default_indices(), graphics::StorageFlag::STATIC_DRAW);
190 va.bind();
191
192 glActiveTexture(GL_TEXTURE0);
193 glBindTexture(GL_TEXTURE_2D, texture);
194
195 rtt->bind();
196 rtt->set_uniform("u_projection", proj);
197 rtt->set_uniform("u_transform", tf.get_transform());
198
199 glDrawElements(GL_TRIANGLES, va.index_count(), GL_UNSIGNED_INT, nullptr);
200 glBindVertexArray(0);
201 glBindTexture(GL_TEXTURE_2D, 0);
202 glUseProgram(0);
203 }
204 else
205 {
206 GALAXY_LOG(GALAXY_ERROR, "Failed to fetch default draw_texture_to_framebuffer shader.");
207 }
