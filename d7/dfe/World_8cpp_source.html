98 if (!components.empty())

184 void World::construct_script(entt::registry& registry, entt::entity entity)
185 {
186 auto& script = registry.get<components::Script>(entity);
187 auto& state = core::ServiceLocator<sol::state>::ref();
188
189 auto result = state.load_file(script.file());
190 if (result.valid())
191 {
192 script.m_self = result.call();
193
194 if (script.m_self.valid())
195 {
196 script.m_update = script.m_self["update"];
197
198 if (!script.m_update.valid())
199 {
200 GALAXY_LOG(GALAXY_ERROR, "Update function not present in '{0}'.", script.file());
201 }
202
203 script.m_self["owner"] = std::ref(registry);
204 script.m_self["dispatcher"] = std::ref(m_scene->m_dispatcher);
205 script.m_self["id"] = sol::readonly_property([entity] {
206 return entity;
207 });
208
209 sol::function init = script.m_self["construct"];
210 if (init.valid())
211 {
212 init(script.m_self);
213 }
214 }
215 else
216 {
217 GALAXY_LOG(GALAXY_ERROR, "Failed to validate script '{0}'. Make sure its in the correct format.", script.file());
218 }
219 }
220 else
221 {
222 GALAXY_LOG(GALAXY_ERROR, "Failed to load script '{0}' because '{1}'.", script.file(), magic_enum::enum_name(result.status()));
223 }
224 }
