125 bool Shader::preprocess(const std::string& src)
126 {
127 auto result = true;
128
129 if (!src.empty())
130 {
131 std::string vertex;
132 std::string frag;
133
134 const auto token = "#type";
135 const auto len = std::strlen(token);
136
137 auto pos = src.find(token, 0);
138 while (pos != std::string::npos)
139 {
140 const auto eol = src.find_first_of("\r\n", pos);
141 const auto begin = pos + len + 1;
142
143 auto type = src.substr(begin, eol - begin);
144
145 if (type == "vertex" || type == "fragment")
146 {
147 const auto next_line = src.find_first_not_of("\r\n", eol);
148 pos = src.find(token, next_line);
149
150 if (type == "vertex")
151 {
152 vertex = (pos == std::string::npos) ? src.substr(next_line) : src.substr(next_line, pos - next_line);
153 }
154 else
155 {
156 frag = (pos == std::string::npos) ? src.substr(next_line) : src.substr(next_line, pos - next_line);
157 }
158 }
159 else
160 {
161 GALAXY_LOG(GALAXY_ERROR, "Failed to parse shader type. Must be 'vertex' or 'fragment'.");
162 result = false;
163 }
164 }
165
166 result = load_into_raylib(vertex, frag);
167 }
168 else
169 {
170 GALAXY_LOG(GALAXY_ERROR, "Empty source code provided to shader.");
171 result = false;
172 }
173
174 return result;
175 }
