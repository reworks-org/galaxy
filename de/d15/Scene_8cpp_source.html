152 void Scene::deserialize(const nlohmann::json& json)
153 {
154 m_camera.deserialize(json.at("camera"));
155 auto& em = core::ServiceLocator<meta::EntityMeta>::ref();
156
157 m_registry.clear();
158
159 const auto& physics = json.at("physics");
160 const auto& gravity = physics.at("gravity");
161
162 m_b2world.SetGravity({gravity.at("x"), gravity.at("y")});
163 m_b2world.SetAllowSleeping(physics.at("allow_sleeping"));
164 m_b2world.SetAutoClearForces(physics.at("allow_autoclearforces"));
165 m_velocity_iterations = physics.at("velocity_iterations");
166 m_position_iterations = physics.at("position_iterations");
167
168 const auto& entity_json = json.at("entities");
169 for (const auto& data : entity_json)
170 {
171 const auto entity = em.deserialize_entity(data, m_registry.m_entt);
172
173 m_registry.m_entt.emplace<flags::Disabled>(entity);
174
175 if (!m_registry.m_entt.all_of<components::Tag>(entity))
176 {
177 auto& tag = m_registry.m_entt.emplace<components::Tag>(entity);
178 tag.m_tag = "Untagged";
179 }
180 }
181
182 m_name = json.at("name");
183
184 if (load_world(json.at("ldtk_world")))
185 {
186 m_world.set_active(json.at("current_map"));
187 }
188 }
