142 void Renderer::draw_texture_to_target(RenderTexture& target, Texture* texture, VertexArray& va, Transform& tf)
143 {
144 glActiveTexture(GL_TEXTURE0);
145
146 target.bind();
147 va.bind();
148 texture->bind();
149
150 // Use a standard projection.
151 const auto proj = glm::ortho(0.0f, static_cast<float>(target.width()), static_cast<float>(target.height()), 0.0f, -1.0f, 1.0f);
152
153 auto rtt = core::ServiceLocator<resource::Shaders>::ref().get("render_to_texture");
154 rtt->bind();
155 rtt->set_uniform("u_projection", proj);
156 rtt->set_uniform("u_transform", tf.get_transform());
157
158 glDrawElements(GL_TRIANGLES, va.count(), GL_UNSIGNED_INT, va.offset());
159
160 glBindVertexArray(0);
161 glBindTexture(GL_TEXTURE_2D, 0);
162 glUseProgram(0);
163 }

165 void Renderer::draw()
166 {
167 // I don't think we would ever sort the amount of renderable data required to do this in parallel faster.
168 std::sort(m_cmds.begin(), m_cmds.end(), [](auto left, auto right) -> bool {
169 return left->layer < right->layer;
170 });
171
172 m_r2d_shader.bind();
173
174 for (auto& cmd : m_cmds)
175 {
176 m_renderdata.sub_buffer<RenderData>(0, 1, &cmd->uniforms);
177
178 glBindVertexArray(cmd->vao);
179 glDrawElementsInstanced(cmd->mode, cmd->count, GL_UNSIGNED_INT, cmd->offset, cmd->instances);
180 }
181 }
