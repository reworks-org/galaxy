268 template<math::Shape shape>
269 inline void Primitive::create(const PrimitiveData& data, const graphics::Colour& colour, const int layer)
270 {
271 m_shape = shape;
272 m_data = data;
273 m_layer = layer;
274
275 meta::vector<unsigned int> indices;
276 meta::vector<graphics::Vertex> vertices;
277
278 if constexpr (shape == math::Shape::CIRCLE)
279 {
280 // Thanks to https://stackoverflow.com/a/33859443
281 // For help with maths.
282
283 auto count = 0u;
284 constexpr const auto incr_stat = 2.0f * glm::pi<float>();
285 const auto increment = incr_stat / data.fragments;
286
287 for (float angle = 0.0f; angle <= (2.0f * glm::pi<float>()); angle += increment)
288 {
289 graphics::Vertex vertex;
290 vertex.m_pos.x = (data.radius * glm::cos(angle)) + data.radius;
291 vertex.m_pos.y = (data.radius * glm::sin(angle) + data.radius);
292
293 vertices.emplace_back(vertex);
294 indices.push_back(count);
295
296 count++;
297 }
298
299 m_mode = graphics::primitive_to_gl(graphics::Primitives::LINE_LOOP);
300 m_width = data.radius * 2.0f;
301 m_height = data.radius * 2.0f;
302 }
303 else if constexpr (shape == math::Shape::ELLIPSE)
304 {
305 // Thanks to https://stackoverflow.com/a/34735255
306 // For help with maths.
307
308 const auto theta = 2.0f * glm::pi<float>() / m_data.fragments;
309 const auto cosine = std::cosf(theta);
310 const auto sine = std::sinf(theta);
311
312 auto temp = 0.0f;
313 auto x = 1.0f;
314 auto y = 0.0f;
315
316 auto count = 0u;
317 for (auto i = 0; i < std::floor(m_data.fragments); i++)
318 {
319 graphics::Vertex vertex;
320 vertex.m_pos.x = (x * m_data.radii.x) + m_data.radii.x;
321 vertex.m_pos.y = (y * m_data.radii.y) + m_data.radii.y;
322
323 vertices.emplace_back(vertex);
324 indices.push_back(count);
325 count++;
326
327 temp = x;
328 x = cosine * x - sine * y;
329 y = sine * temp + cosine * y;
330 }
331
332 m_mode = graphics::primitive_to_gl(graphics::Primitives::LINE_LOOP);
333 m_width = m_data.radii.x * 2.0f;
334 m_height = m_data.radii.y * 2.0f;
335 }
336 else if constexpr (shape == math::Shape::LINE)
337 {
338 graphics::Vertex a;
339 a.m_pos.x = m_data.start_end.x;
340 a.m_pos.y = m_data.start_end.y;
341
342 graphics::Vertex b;
343 b.m_pos.x = m_data.start_end.z;
344 b.m_pos.y = m_data.start_end.w;
345
346 vertices.emplace_back(a);
347 vertices.emplace_back(b);
348 indices.push_back(0);
349 indices.push_back(1);
350
351 m_mode = graphics::primitive_to_gl(graphics::Primitives::LINE);
352 m_width = 0.0f;
353 m_height = 0.0f;
354 }
355 else if constexpr (shape == math::Shape::POINT)
356 {
357 graphics::Vertex vertex;
358 vertex.m_pos = {0.0f, 0.0f};
359
360 vertices.emplace_back(vertex);
361 indices.push_back(0);
362
363 m_mode = graphics::primitive_to_gl(graphics::Primitives::POINT);
364 m_width = 0.0f;
365 m_height = 0.0f;
366 }
367 else if constexpr ((shape == math::Shape::POLYLINE) || (shape == math::Shape::POLYGON))
368 {
369 m_width = 0.0f;
370 m_height = 0.0f;
371
372 auto count = 0u;
373 for (const auto& point : m_data.points)
374 {
375 if constexpr (shape == math::Shape::POLYGON)
376 {
377 if (point.x > m_width)
378 {
379 m_width = point.x;
380 }
381
382 if (point.y > m_height)
383 {
384 m_height = point.y;
385 }
386 }
387
388 graphics::Vertex vertex;
389 vertex.m_pos.x = point.x;
390 vertex.m_pos.y = point.y;
391
392 vertices.emplace_back(vertex);
393 indices.push_back(count);
394
395 count++;
396 }
397
398 m_mode = graphics::primitive_to_gl(graphics::Primitives::LINE_LOOP);
399 }
400
401 m_vao.create(vertices, graphics::StorageFlag::STATIC_DRAW, indices, graphics::StorageFlag::STATIC_DRAW);
402 }
