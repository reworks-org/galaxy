100 // Update and Render additional Platform Windows
101 // (Platform functions may change the current OpenGL context, so we save/restore it to make it easier to paste this code elsewhere.

103 const auto& io = ImGui::GetIO();

105 {
106 GLFWwindow* backup_current_context = glfwGetCurrentContext();

127 void imgui_confirm(const char* popup, const std::function<void(void)>& yes, const std::function<void(void)>& no)
128 {
129 if (popup_state[popup])
130 {
131 ImGui::OpenPopup(popup);
132 imgui_center_next_window();
133
134 popup_state[popup] = false;
135 }
136
137 if (ImGui::BeginPopup(popup))
138 {
139 ImGui::Text("Are you sure you want to?");
140
141 ImGui::Separator();
142 ImGui::Spacing();
143
144 if (ImGui::Button("Yes"))
145 {
146 if (yes)
147 {
148 yes();
149 }
150
151 ImGui::CloseCurrentPopup();
152 }
153
154 ImGui::SameLine();
155
156 if (ImGui::Button("No"))
157 {
158 if (no)
159 {
160 no();
161 }
162
163 ImGui::CloseCurrentPopup();
164 }
165
166 ImGui::EndPopup();
167 }
