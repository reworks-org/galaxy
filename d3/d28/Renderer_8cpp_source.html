62 void Renderer::draw()
63 {
64 s_prev_texture = -1;
65 s_prev_nm = -1;
66
67 // I don't think we would ever sort the amount of renderable data required to do this in parallel faster.
68 // std::execution::par maybe used when > 1000? But 1000 draw calls is too slow anyway.
69
70 std::sort(s_data.begin(), s_data.end(), [](const RenderCommand& left, const RenderCommand& right) -> bool {
71 if (left.renderable->get_layer() == right.renderable->get_layer())
72 {
73 return left.renderable->get_texture_handle() < right.renderable->get_texture_handle();
74 }
75 else
76 {
77 return left.renderable->get_layer() < right.renderable->get_layer();
78 }
79 });
80
81 s_r2d_shader.bind();
82
83 for (auto& cmd : s_data)
84 {
85 s_r2d_ubo->buffer<Render2DUniform>(0, 1, &cmd.uniform_data);
86
87 glBindVertexArray(cmd.renderable->get_vao().id());
88
89 const auto tex = cmd.renderable->get_texture_handle();
90 if (s_prev_texture != tex)
91 {
92 glBindTextureUnit(0, tex);
93 s_prev_texture = tex;
94 }
95
96 // Instances = 1 is the same as glDrawElements.
97 glDrawElementsInstanced(cmd.mode, cmd.renderable->get_vao().index_count(), GL_UNSIGNED_INT, nullptr, cmd.instances);
98 }

132 void Renderer::draw_texture(const unsigned int texture, const int width, const int height)
133 {
134 auto rtt = core::ServiceLocator<resource::Shaders>::ref().get("RenderToTexture");
135 if (rtt != nullptr)
136 {
137 VertexArray va;
138 components::Transform tf;
139

141
142 auto v = graphics::Vertex::gen_quad_vertices(width, height);
143 va.create(v, graphics::StorageFlag::STATIC_DRAW, graphics::Vertex::get_default_indices(), graphics::StorageFlag::STATIC_DRAW);
144 va.bind();
145
146 glActiveTexture(GL_TEXTURE0);
147 glBindTexture(GL_TEXTURE_2D, texture);
148
149 rtt->bind();
150 rtt->set_uniform("u_projection", proj);
151 rtt->set_uniform("u_transform", tf.get_transform());
152
153 glDrawElements(GL_TRIANGLES, va.index_count(), GL_UNSIGNED_INT, nullptr);
154 glBindVertexArray(0);
155 glBindTexture(GL_TEXTURE_2D, 0);
