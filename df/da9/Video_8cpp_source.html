27 void data_callback(ma_device* pDevice, void* pOutput, const void* pInput, ma_uint32 frameCount)
28 {
29 auto buffer = static_cast<RingBuffer*>(pDevice->pUserData);
30
31 const int frames_needed = frameCount * 2;
32 if (buffer->available_bytes() >= frames_needed)
33 {
34 const float* read = buffer->direct_read_pointer(frames_needed);
35 auto output = static_cast<float*>(pOutput);
36
37 for (auto i = 0; i < frames_needed; i++)
38 {
39 output[i] = read[i] * buffer->get_volume();
40 }
41 }
42 else
43 {
44 std::memset(pOutput, 0, frames_needed * sizeof(float));
45 }
46 }

101 bool Video::load(const std::string& file)
102 {
103 auto success = true;
104
105 auto& fs = core::ServiceLocator<fs::VirtualFileSystem>::ref();
106 m_buffer = fs.read_binary(file);
107 if (!m_buffer.empty())
108 {
109 m_plm = plm_create_with_memory(m_buffer.data(), m_buffer.size(), false);
110 if (m_plm)
111 {
112 plm_set_video_decode_callback(
113 m_plm,
114 [](plm_t* self, plm_frame_t* frame, void* user) {
115 auto video = static_cast<Video*>(user);
116
117 video->update_texture(GL_TEXTURE0, video->m_texture_y, &frame->y);
118 video->update_texture(GL_TEXTURE1, video->m_texture_cb, &frame->cb);
119 video->update_texture(GL_TEXTURE2, video->m_texture_cr, &frame->cr);
120 },
121 this);
122
123 plm_set_audio_decode_callback(
124 m_plm,
125 [](plm_t* self, plm_samples_t* samples, void* user) {
126 auto buffer = static_cast<RingBuffer*>(user);
127 buffer->write(samples->interleaved, samples->count * 2);
128 },
129 m_audio_buffer.get());
130
131 if (plm_get_num_audio_streams(m_plm) > 0)
132 {
133 m_audio_buffer = std::make_unique<RingBuffer>(plm_get_samplerate(m_plm));
134
135 ma_device_config config = ma_device_config_init(ma_device_type_playback);
136 config.playback.format = ma_format_f32;
137 config.playback.channels = 2;
138 config.sampleRate = plm_get_samplerate(m_plm);
139 config.pUserData = m_audio_buffer.get();
140 config.dataCallback = data_callback;
141
142 if (ma_device_init(nullptr, &config, m_audio) != MA_SUCCESS)
143 {
144 plm_set_audio_enabled(m_plm, false);
145 plm_set_loop(m_plm, false);
146 plm_set_audio_stream(m_plm, 0);
147 plm_set_audio_enabled(m_plm, true);
148
149 if (!m_shader.parse(resource::video_vert_shader, resource::video_frag_shader))
150 {
151 GALAXY_LOG(GALAXY_ERROR, "Failed to load plm video shader '{0}'.", file);
152 success = false;
153 }
154 }

158 plm_set_audio_enabled(m_plm, false);
159 }
160 }

164 success = false;
165 }
166 }

170 success = false;

270 {
271 unsigned int texture = 0;
272 glCreateTextures(GL_TEXTURE_2D, 1, &texture);
273
274 glBindTexture(GL_TEXTURE_2D, texture);
275 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
276 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
277 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
278 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
279
280 constexpr const int swizzleMask[] = {GL_ZERO, GL_ZERO, GL_ZERO, GL_RED};
281 glTexParameteriv(GL_TEXTURE_2D, GL_TEXTURE_SWIZZLE_RGBA, swizzleMask);
282
283 glBindTexture(GL_TEXTURE_2D, 0);
284
285 m_shader.set_uniform(uniform, index);
286
287 return texture;
