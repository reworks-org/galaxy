31local function _createIndexWrapper(aClass, f)
32 if f == nil then
33 return aClass.__instanceDict
34 elseif type(f) == "function" then
35 return function(self, name)
36 local value = aClass.__instanceDict[name]
37
38 if value ~= nil then
39 return value
40 else
41 return (f(self, name))
42 end
43 end
44 else -- if type(f) == "table" then
45 return function(self, name)
46 local value = aClass.__instanceDict[name]
47

49 return value

109local function _includeMixin(aClass, mixin)
110 assert(type(mixin) == 'table', "mixin must be a table")
111
112 for name,method in pairs(mixin) do
113 if name ~= "included" and name ~= "static" then aClass[name] = method end
114 end
115
116 for name,method in pairs(mixin.static or {}) do
117 aClass.static[name] = method
118 end
119
120 if type(mixin.included)=="function" then mixin:included(aClass) end
121 return aClass
122end
